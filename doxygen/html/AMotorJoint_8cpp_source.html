103 pgd::Vector3 AMotorJoint::GetEulerAngles(const Marker &basisMarker) const
104 {
105  // returns the Euler angles using marker axes as the basis
106 
107  // Theory:
108  // Since the axes form an orthonormal basis A (each axis is a column of the matrix),
109  // you can decompose a rotation R on the three axes by transforming R into the
110  // basis A and then do Euler Angle decomposition on the three main axes:
111  // R = A*R'*A^t = A*X*Y*Z*A^t = (A*X*A^t)*(A*Y*A^t)*(A*Z*A^t)
112  // This translates into the following algorithm:
113  // Compute R' = A^t*R*A
114  // Decompose R' into Euler Angles around main axes to obtain angles X, Y, Z
115  // convert these into their matrix rotations
116  // Compute the three rotations around the given axes:
117  // X' = A*X*A^t
118  // Y' = A*Y*A^t
119  // Z' = A*Y*A^t
120 
121  pgd::Quaternion body1MarkerWorld = body1Marker()->GetWorldQuaternion();
122  pgd::Quaternion body2MarkerWorld = body2Marker()->GetWorldQuaternion();
123  pgd::Matrix3x3 R;
124  if (m_reverseBodyOrderInCalculations)
125  {
126  // angle is calculated with respect to body 1
127  pgd::Quaternion body1ToBody2 = body2MarkerWorld * (~body1MarkerWorld);
128  R = pgd::Matrix3x3(body1ToBody2);
129  }
130  else
131  {
132  // angle is calculated with respect to body 2
133  pgd::Quaternion body2ToBody1 = body1MarkerWorld * (~body2MarkerWorld);
134  R = pgd::Matrix3x3(body2ToBody1);
135  }
136  pgd::Matrix3x3 A = pgd::MakeMFromQ(basisMarker.GetWorldQuaternion());
137  pgd::Matrix3x3 At = A.Transpose();
138  pgd::Matrix3x3 Rp = At * R * A;
139  pgd::Vector3 euler = pgd::MakeEulerAnglesFromQRadian(pgd::MakeQfromM(Rp));
140  pgd::Matrix3x3 Xp = A * pgd::MakeMFromQ(pgd::MakeQFromAxisAngle(1, 0, 0, (euler.x), true)) * At;
141  pgd::Matrix3x3 Yp = A * pgd::MakeMFromQ(pgd::MakeQFromAxisAngle(0, 1, 0, (euler.y), true)) * At;
142  pgd::Matrix3x3 Zp = A * pgd::MakeMFromQ(pgd::MakeQFromAxisAngle(0, 0, 1, (euler.z), true)) * At;
143  pgd::Quaternion Xpq = pgd::MakeQfromM(Xp);
144  pgd::Quaternion Ypq = pgd::MakeQfromM(Yp);
145  pgd::Quaternion Zpq = pgd::MakeQfromM(Zp);
146 
147  // we can't just convert the quaternions into axis angle to get the angles because the axes might be reversed, or might be arbitrary if close to zero
148  double xMag = Xpq.GetVector().Magnitude();
149  double yMag = Ypq.GetVector().Magnitude();
150  double zMag = Zpq.GetVector().Magnitude();
151  double dotX = 0, dotY = 0, dotZ = 0;
152  pgd::Vector3 axisX, axisY, axisZ;
153  double angle0 = 0, angle1 = 0, angle2 = 0;
154  const double epsilon = std::numeric_limits<double>::epsilon();
155  if (xMag < epsilon || Xpq.n >= 1 || Xpq.n <= -1)
156  {
157  angle0 = 0;
158  }
159  else
160  {
161  angle0 = 2 * std::acos(Xpq.n); // acos produces a value from 0 to pi, so this angle is 0 to 2*pi
162  if (angle0 > M_PI) angle0 -= 2 * M_PI; // so this means the angle is restricted to the more usual -pi to pi
163  axisX = Xpq.GetVector() / xMag;
164  dotX =pgd::Dot(axisX, pgd::Vector3(A.e11, A.e21, A.e31));
165  if (dotX < 0)
166  angle0 = - angle0;
167  }
168  if (yMag < epsilon || Ypq.n >= 1 || Ypq.n <= -1)
169  {
170  angle1 = 0;
171  }
172  else
173  {
174  angle1 = 2 * std::acos(Ypq.n);
175  if (angle1 > M_PI) angle1 -= 2 * M_PI;
176  axisY = Ypq.GetVector() / yMag;
177  dotY =pgd::Dot(axisY, pgd::Vector3(A.e11, A.e21, A.e31));
178  if (dotY < 0)
179  angle1 = - angle1;
180  }
181  if (zMag < epsilon || Zpq.n >= 1 || Zpq.n <= -1)
182  {
183  angle2 = 0;
184  }
185  else
186  {
187  angle2 = 2 * std::acos(Zpq.n);
188  if (angle2 > M_PI) angle2 -= 2 * M_PI;
189  axisZ = Zpq.GetVector() / zMag;
190  dotZ =pgd::Dot(axisZ, pgd::Vector3(A.e11, A.e21, A.e31));
191  if (dotZ < 0)
192  angle2 = - angle2;
193  }
194  return pgd::Vector3(angle0, angle1, angle2);
195 }

298 void AMotorJoint::Update()
299 {
300  // handle the angular change code
301  m_currentQuaternion = GetQuaternion();
302  if (m_firstTime) m_lastQuaternion = m_currentQuaternion;
303  m_lastToCurrent = m_currentQuaternion * (~m_lastQuaternion);
304  m_lastQuaternion = m_currentQuaternion;
305 
306  pgd::Quaternion targetQuaternion = pgd::MakeQFromAxisAngle(m_targetAxis, m_targetAngle);
307  pgd::Quaternion currentToTarget = targetQuaternion * (~m_currentQuaternion);
308  pgd::MakeAxisAngleFromQ(currentToTarget, &m_deltaAxis.x, &m_deltaAxis.y, &m_deltaAxis.z, &m_deltaAngle);
309  if (GetBody1())
310  {
311  pgd::Quaternion bodyRotation(GetBody1()->GetQuaternion());
312  m_deltaAxis = pgd::QVRotate(bodyRotation, m_deltaAxis);
313  }
314  if (m_firstTime) m_lastDeltaAxis = m_deltaAxis;
315  if (pgd::Dot(m_lastDeltaAxis, m_deltaAxis) < 0)
316  {
317  m_deltaAxis = -m_deltaAxis;
318  m_deltaAngle = -m_deltaAngle;
319  }
320  dJointSetAMotorAxis(JointID(), 0, 0, m_deltaAxis.x, m_deltaAxis.y, m_deltaAxis.z);
321  double targetVelocity = m_deltaAngle * m_targetAngleGain;
322  if (targetVelocity * simulation()->GetTimeIncrement() > m_deltaAngle) targetVelocity = 0.5 * m_deltaAngle / simulation()->GetTimeIncrement();
323  dJointSetAMotorParam(JointID(), dParamVel1, targetVelocity);
324 
325  m_firstTime = false;
326 }
