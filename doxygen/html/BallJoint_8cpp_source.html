242 std::string *BallJoint::createFromAttributes()
243 {
244  if (Joint::createFromAttributes()) return lastErrorPtr();
245  std::string buf;
246  if (findAttribute("Mode"s, &buf) == nullptr) return lastErrorPtr();
247  else if (buf == "NoStops") m_Mode = BallJoint::NoStops;
248  else if (buf == "AMotorUser") m_Mode = BallJoint::AMotorUser;
249  else if (buf == "AMotorEuler") m_Mode = BallJoint::AMotorEuler;
250  else { setLastError("Joint ID=\""s + name() +"\" unrecognised Mode"s); return lastErrorPtr(); }
251 
252  pgd::Vector3 position = body1Marker()->GetWorldPosition();
253  this->SetBallAnchor(position.x, position.y, position.z);
254  pgd::Vector3 x, y, z;
255  body1Marker()->GetWorldBasis(&x, &y, &z);
256  int axisMode = 1; // 1 relative to body 1
257  this->SetAxes(x.x, x.y, x.z, y.x, y.y, y.z, z.x, z.y, z.z, axisMode);
258  if (CFM() >= 0) dJointSetBallParam (JointID(), dParamCFM, CFM());
259  if (ERP() >= 0) dJointSetBallParam (JointID(), dParamERP, ERP());
260 
261  switch (m_Mode)
262  {
263  case BallJoint::NoStops:
264  break;
265  case BallJoint::AMotorUser:
266  {
267  if (findAttribute("LowStop0"s, &buf) == nullptr) return lastErrorPtr();
268  double a0Low = GSUtil::GetAngle(buf);
269  if (findAttribute("HighStop0"s, &buf) == nullptr) return lastErrorPtr();
270  double a0High = GSUtil::GetAngle(buf);
271  if (findAttribute("LowStop1"s, &buf) == nullptr) return lastErrorPtr();
272  double a1Low = GSUtil::GetAngle(buf);
273  if (findAttribute("HighStop1"s, &buf) == nullptr) return lastErrorPtr();
274  double a1High = GSUtil::GetAngle(buf);
275  if (findAttribute("LowStop2"s, &buf) == nullptr) return lastErrorPtr();
276  double a2Low = GSUtil::GetAngle(buf);
277  if (findAttribute("HighStop2"s, &buf) == nullptr) return lastErrorPtr();
278  double a2High = GSUtil::GetAngle(buf);
279  if (a0Low >= a0High) { setLastError("Joint ID=\""s + name() +"\" LowStop0 >= HighStop0"s); return lastErrorPtr(); }
280  if (a1Low >= a1High) { setLastError("Joint ID=\""s + name() +"\" LowStop1 >= HighStop1"s); return lastErrorPtr(); }
281  if (a2Low >= a2High) { setLastError("Joint ID=\""s + name() +"\" LowStop2 >= HighStop2"s); return lastErrorPtr(); }
282  this->SetStops(a0Low, a0High, a1Low, a1High, a2Low, a2High);
283  break;
284  }
285  case BallJoint::AMotorEuler:
286  {
287  if (findAttribute("LowStop0"s, &buf) == nullptr) return lastErrorPtr();
288  double a0Low = GSUtil::GetAngle(buf);
289  if (findAttribute("HighStop0"s, &buf) == nullptr) return lastErrorPtr();
290  double a0High = GSUtil::GetAngle(buf);
291  if (findAttribute("LowStop1"s, &buf) == nullptr) return lastErrorPtr();
292  double a1Low = GSUtil::GetAngle(buf);
293  if (findAttribute("HighStop1"s, &buf) == nullptr) return lastErrorPtr();
294  double a1High = GSUtil::GetAngle(buf);
295  if (findAttribute("LowStop2"s, &buf) == nullptr) return lastErrorPtr();
296  double a2Low = GSUtil::GetAngle(buf);
297  if (findAttribute("HighStop2"s, &buf) == nullptr) return lastErrorPtr();
298  double a2High = GSUtil::GetAngle(buf);
299  if (a0Low >= a0High) { setLastError("Joint ID=\""s + name() +"\" LowStop0 >= HighStop0"s); return lastErrorPtr(); }
300  if (a1Low >= a1High) { setLastError("Joint ID=\""s + name() +"\" LowStop1 >= HighStop1"s); return lastErrorPtr(); }
301  if (a2Low >= a2High) { setLastError("Joint ID=\""s + name() +"\" LowStop2 >= HighStop2"s); return lastErrorPtr(); }
302  this->SetStops(a0Low, a0High, a1Low, a1High, a2Low, a2High);
303  break;
304  }
305  }
306  return nullptr;
307 }
