67 // parses the position allowing a relative position specified by BODY ID
68 // x y z - world coordinates
69 // bodyName x y z - position relative to bodyName local coordinate system
70 // bodyName x1 y1 z1 x2 y2 z2 - position such that x1,y1,z1 on bodyName has same world coordinates as x2,y2,z2 on local body
71 std::string *Body::SetPosition(const std::string &buf)
72 {
73  std::vector<std::string> tokens;
74  pystring::split(buf, tokens);
75 
76  if (tokens.size() == 1) // it must be a marker
77  {
78  Marker *marker = simulation()->GetMarker(tokens[0]);
79  if (marker == nullptr)
80  {
81  setLastError("Body ID=\""s + name() +"\" Position=\""s + buf + "\" marker not found"s);
82  return lastErrorPtr();
83  }
84  pgd::Vector3 wp = marker->GetWorldPosition();
85  this->SetPosition(wp.x, wp.y, wp.z);
86  return nullptr;
87  }
88 
89  if (tokens.size() == 2) // two marker definition
90  {
91  Marker *marker1 = simulation()->GetMarker(tokens[0]);
92  Marker *marker2 = simulation()->GetMarker(tokens[1]);
93  if (marker1 == nullptr || marker2 == nullptr)
94  {
95  setLastError("Body ID=\""s + name() +"\" Position=\""s + buf + "\" markers not found"s);
96  return lastErrorPtr();
97  }
98  // find the marker that is relative to this body and make it marker1
99  if (marker2->GetBody()->GetBodyID() == m_bodyID) std::swap(marker1, marker2);
100  if (marker2->GetBody()->GetBodyID() == m_bodyID)
101  {
102  setLastError("Body ID=\""s + name() +"\" Position=\""s + buf + "\" Only one marker of pair must be relative to the body"s);
103  return lastErrorPtr();
104  }
105  if (marker1->GetBody()->GetBodyID() != m_bodyID)
106  {
107  setLastError("Body ID=\""s + name() +"\" Position=\""s + buf + "\" Only one marker of pair must be relative to the body"s);
108  return lastErrorPtr();
109  }
110 
111  pgd::Vector3 target = marker2->GetWorldPosition();
112  pgd::Vector3 current = marker2->GetWorldPosition();
113  pgd::Vector3 difference = target - current;
114  this->SetPosition(difference.x, difference.y, difference.z);
115  return nullptr;
116  }
117 
118  if (tokens.size() == 3)
119  {
120  this->SetPosition(GSUtil::Double(tokens[0]), GSUtil::Double(tokens[1]), GSUtil::Double(tokens[2]));
121  return nullptr;
122  }
123 
124  if (tokens.size() == 4)
125  {
126  Body *theBody = simulation()->GetBody(tokens[0]);
127  if (theBody == nullptr)
128  {
129  if (tokens[0] == "World"s)
130  {
131  this->SetPosition(GSUtil::Double(tokens[1]), GSUtil::Double(tokens[2]), GSUtil::Double(tokens[3]));
132  return nullptr;
133  }
134  else
135  {
136  setLastError("Body ID=\""s + name() +"\" Position=\""s + buf + "\" reference body not found"s);
137  return lastErrorPtr();
138  }
139  }
140  else
141  {
142  dVector3 result;
143  dBodyGetRelPointPos (theBody->GetBodyID(), GSUtil::Double(tokens[1]), GSUtil::Double(tokens[2]), GSUtil::Double(tokens[3]), result);
144  this->SetPosition(result[0], result[1], result[2]);
145  return nullptr;
146  }
147  }
148  if (tokens.size() == 7)
149  {
150  Body *theBody = simulation()->GetBody(tokens[0]);
151  if (!theBody)
152  {
153  setLastError("Body ID=\""s + name() +"\" Position=\""s + buf + "\" reference body not found"s);
154  return lastErrorPtr();
155  }
156  // get world coordinates of x1,y1,z1
157  dVector3 world1, world2, pos;
158  dBodyGetRelPointPos (theBody->GetBodyID(), GSUtil::Double(tokens[1]), GSUtil::Double(tokens[2]), GSUtil::Double(tokens[3]), world1);
159  dBodyGetRelPointPos (m_bodyID, GSUtil::Double(tokens[4]), GSUtil::Double(tokens[5]), GSUtil::Double(tokens[6]), world2);
160  // add the error to the current position
161  const double *p = dBodyGetPosition(m_bodyID);
162  for (size_t i = 0; i < 3; i++) pos[i] = p[i] + (world1[i] - world2[i]);
163  this->SetPosition(pos[0], pos[1], pos[2]);
164 
165  // for checking
166  // dBodyGetRelPointPos (m_BodyID, local2[0], local2[1], local2[2], world2);
167  // for (i = 0; i < 3; i++) std::cerr << world1[i] << " " << world2[i] << "\n";
168  return nullptr;
169  }
170 
171  setLastError("Body ID=\""s + name() +"\" Position=\""s + buf + "\" too many tokens"s);
172  return lastErrorPtr();
173 }

335 double Body::GetRotationalKineticEnergy()
336 {
337 
338  // rotational KE = 0.5 * o(t) * I * o
339  // where o is rotational velocity vector and o(t) is the same but transposed
340 
341  dMass mass;
342  dBodyGetMass(m_bodyID, &mass);
343 
344  const double *ow = dBodyGetAngularVel(m_bodyID);
345  dVector3 o;
346  dBodyVectorFromWorld (m_bodyID, ow[0], ow[1], ow[2], o);
347  dVector3 o1;
348  dMultiply0_331(o1, mass.I, o);
349  double rotationalKE = 0.5 * (o[0]*o1[0] + o[1]*o1[1] + o[2]*o1[2]);
350 
351  return rotationalKE;

574 double Body::GetProjectedAngle(const pgd::Vector3 &planeNormal, const pgd::Vector3 &vector1, const pgd::Vector3 &vector2)
575 {
576  // calculate the projected angle from vector1 to vector2 projected onto the plane defined by a normal vector
577  // this angle gives the proper -pi to +pi value to rotate the direction for vector1 to vector2
578  // around planeNormal
579 
580  // start by projecting the two vectors onto a plane defined by the first axis as a normal
581  // from https://www.euclideanspace.com/maths/geometry/elements/plane/lineOnPlane/index.htm
582  // using the formula B cross (A cross B / |B|) / |B|
583  // where A is the vector and B is the normal of the plane
584  // for completeness this distance along the normal is
585  // A dot B * B/|B|**2
586  // however we can use some optimisation because we just want the direction in the plane which is
587  // B cross (A cross B)
588  // which we then normalise anyway
589  pgd::Vector3 startDirection1 = planeNormal ^ (vector1 ^ planeNormal);
590  startDirection1.Normalize();
591  pgd::Vector3 endDirection1 = planeNormal ^ (vector2 ^ planeNormal);
592  endDirection1.Normalize();
593  // now we can find the angle in the plane using

778 std::string Body::MassCheck(const dMass *m)
779 {
780  if (m->mass <= 0)
781  {
782  return "mass must be > 0"s;
783  }
784  if (!dIsPositiveDefinite (m->I,3))
785  {
786  return "inertia must be positive definite"s;
787  }
788 
789  // verify that the center of mass position is consistent with the mass
790  // and inertia matrix. this is done by checking that the inertia around
791  // the center of mass is also positive definite. from the comment in
792  // dMassTranslate(), if the body is translated so that its center of mass
793  // is at the point of reference, then the new inertia is:
794  // I + mass*crossmat(c)^2
795  // note that requiring this to be positive definite is exactly equivalent
796  // to requiring that the spatial inertia matrix
797  // [ mass*eye(3,3) M*crossmat(c)^T ]
798  // [ M*crossmat(c) I ]
799  // is positive definite, given that I is PD and mass>0. see the theorem
800  // about partitioned PD matrices for proof.
801 
802  dMatrix3 I2,chat;
803  dSetZero (chat,12);
804  dSetCrossMatrixPlus (chat,m->c,4);
805  dMultiply0_333 (I2,chat,chat);
806  for (int i=0; i<3; i++) I2[i] = m->I[i] + m->mass*I2[i];
807  for (int i=4; i<7; i++) I2[i] = m->I[i] + m->mass*I2[i];
808  for (int i=8; i<11; i++) I2[i] = m->I[i] + m->mass*I2[i];
809  if (!dIsPositiveDefinite (I2,3))
810  {
811  return "center of mass inconsistent with mass parameters"s;
812  }
813  return ""s;
