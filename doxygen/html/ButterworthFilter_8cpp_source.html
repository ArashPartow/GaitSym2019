83 void ButterworthFilter::CalculateCoefficients(double cutoffFrequency, double samplingFrequency)
84 {
85  m_cutoffFrequency = cutoffFrequency;
86  m_samplingFrequency = samplingFrequency;
87  if (m_samplingFrequency <= 0 || m_cutoffFrequency <= 0) // turn off the filter
88  {
89  m_b0 = 1;
90  m_b1 = 0;
91  m_b2 = 0;
92  m_a1 = 0;
93  m_a2 = 0;
94  }
95  // calculate the 2nd Order Butterworth Low Pass Filter coefficients for the IIR filter
96  // y(n) = b0*x(n) + b1*x(n-1) + b2*x(n-2) + a1*y(n-1) + a2*y(n-2)
97  double ff = m_cutoffFrequency / m_samplingFrequency;
98  const double ita = 1.0 / tan(M_PI * ff);
99  const double q = M_SQRT2;
100  m_b0 = 1.0 / (1.0 + q*ita + ita*ita);
101  m_b1 = 2 * m_b0;
102  m_b2 = m_b0;
103  m_a1 = 2.0 * (ita * ita - 1.0) * m_b0;
104  m_a2 = -(1.0 - q * ita + ita * ita) * m_b0;
105 }
