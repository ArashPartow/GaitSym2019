30 std::string *ConvexGeom::createFromAttributes()
31 {
32  if (Geom::createFromAttributes()) return lastErrorPtr();
33  std::string buf;
34  buf.reserve(1000000);
35  if (findAttribute("IndexStart"s, &buf) == nullptr) return lastErrorPtr();
36  m_indexStart = GSUtil::Int(buf);
37  if (findAttribute("Vertices"s, &buf) == nullptr) return lastErrorPtr();
38  GSUtil::Double(buf, &m_vertices);
39  if (findAttribute("Triangles"s, &buf) == nullptr) return lastErrorPtr();
40  GSUtil::Int(buf, &m_triangles);
41  if (findAttribute("ReverseWinding"s, &buf)) m_reverseWinding = GSUtil::Bool(buf);
42  if (m_indexStart) { for (size_t i = 0; i < m_triangles.size(); i++) { m_triangles[i] -= m_indexStart; } }
43  if (m_reverseWinding) { for (size_t i = 0; i < m_triangles.size(); i += 3) { std::swap(m_triangles[i], m_triangles[i + 2]); } }
44  initialiseConvexData();
45  setConvex(m_planes.data(), m_planecount, m_points.data(), m_pointcount, m_polygons.data());
46  return nullptr;
47 }

60  setAttribute("Triangles"s, *GSUtil::ToString(m_triangles.data(), m_triangles.size(), &buf));
61  if (m_reverseWinding) { for (size_t i = 0; i < m_triangles.size(); i += 3) { std::swap(m_triangles[i], m_triangles[i + 2]); } }
62  if (m_indexStart) { for (size_t i = 0; i < m_triangles.size(); i++) { m_triangles[i] -= m_indexStart; } }

72 void ConvexGeom::initialiseConvexData()
73 {
74  // set the limits
75  m_pointcount = static_cast<unsigned int>(m_vertices.size() / 3);
76  m_planecount = static_cast<unsigned int>(m_triangles.size() / 3);
77  // copy the vertices into m_points
78  m_points.clear();
79  m_points.reserve(m_pointcount * 3);
80  for (size_t i = 0; i < m_vertices.size(); i++) m_points.push_back(m_vertices[i]);
81  // copy the triangles into m_polygons with the correct vertex count
82  m_polygons.clear();
83  m_polygons.reserve(m_planecount * 4);
84  for (size_t i = 0; i < m_triangles.size();)
85  {
86  m_polygons.push_back(3);
87  m_polygons.push_back(m_triangles[i++]);
88  m_polygons.push_back(m_triangles[i++]);
89  m_polygons.push_back(m_triangles[i++]);
90  }
91  // calculate and write the planes
92  // Transform a parametric plane form to the cartesian form
93  //
94  // We are given a plane in the parametric form x = p + ru + sv and
95  // want to transform it to the cartesian form ax+ by + cz = d.
96  // First we need to calculate the normal vector equation of the plane by using the cross product:
97  //
98  // n = cross(u, v)
99  //
100  // n may need normalising depending on how u and v are calculated
101  // We calculate d as dot(n, p) and a, b, and c are the components of the n vector:
102  //
103  // dot(n, x) = dot(n, p)
104  // n1x + n2y + n3z = dot(n, p)
105  // ax + by + cz = d
106  m_planes.clear();
107  m_planes.reserve((m_triangles.size() / 3) * 4);
108  pgd::Vector3 v1, v2, v3;
109  pgd::Vector3 p, u, v;
110  pgd::Vector3 n;
111  double d;
112  for (size_t i = 0; i < m_triangles.size();)
113  {
114  v1.Set(m_vertices[m_triangles[i] * 3], m_vertices[m_triangles[i] * 3 + 1], m_vertices[m_triangles[i] * 3 + 2]);
115  i++;
116  v2.Set(m_vertices[m_triangles[i] * 3], m_vertices[m_triangles[i] * 3 + 1], m_vertices[m_triangles[i] * 3 + 2]);
117  i++;
118  v3.Set(m_vertices[m_triangles[i] * 3], m_vertices[m_triangles[i] * 3 + 1], m_vertices[m_triangles[i] * 3 + 2]);
119  i++;
120  // for a triangle v1, v2, v3, if the vector U = v2 - v1 and the vector V = v3 - v1
121  // then the normal N = U x V
122  // Note: the normal is the same if V = v3 - v2
123  p = v1;
124  u = v2 - v1;
125  v = v3 - v1;
126  n = pgd::Cross(u, v);
127  n.Normalize();
128  d = pgd::Dot(n, p);
129  m_planes.push_back(n.x);
130  m_planes.push_back(n.y);
131  m_planes.push_back(n.z);
132  m_planes.push_back(d);
133  // std::cerr << n.x << " " << n.y << " " << n.z << " " << d << "\n";
134  }
135 }
