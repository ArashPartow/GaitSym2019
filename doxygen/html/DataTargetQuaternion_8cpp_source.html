63 double DataTargetQuaternion::calculateError(double time)
64 {
65  const double *r;
66  Body *body;
67  Geom *geom;
68  dQuaternion q;
69  double angle = 0;
70 
71  size_t index, indexNext;
72  auto lowerBound = std::lower_bound(targetTimeList()->begin(), targetTimeList()->end(), time);
73  auto upperBound = std::upper_bound(targetTimeList()->begin(), targetTimeList()->end(), time);
74  if (lowerBound == targetTimeList()->begin()) // time <= lowest value in the list
75  {
76  index = 0;
77  indexNext = index;
78  }
79  else if (upperBound == targetTimeList()->end()) // time > highest value in the list
80  {
81  index = targetTimeList()->size() - 1;
82  indexNext = index;
83  }
84  else
85  {
86  index = std::distance(targetTimeList()->begin(), lowerBound) - 1; // subtracting 1 because lower bound gives an index 1 higher than expected (IMO)
87  indexNext = std::min(index + 1, targetTimeList()->size() - 1);
88  }
89 
90  // do a slerp interpolation between the target quaternions
91  double delTime = (*targetTimeList())[size_t(indexNext)] - (*targetTimeList())[size_t(index)];
92  double interpolationFraction;
93  if (delTime < DBL_EPSILON) interpolationFraction = 0;
94  else interpolationFraction = (time - (*targetTimeList())[size_t(index)]) / delTime;
95  pgd::Quaternion interpolatedTarget = pgd::slerp(m_QValueList[size_t(index)], m_QValueList[size_t(indexNext)], interpolationFraction);
96 
97  if ((body = dynamic_cast<Body *>(GetTarget())) != nullptr)
98  {
99  r = body->GetQuaternion();
100  angle = pgd::FindAngle(interpolatedTarget, pgd::Quaternion(r[0], r[1], r[2], r[3]));
101  }
102  else if ((geom = dynamic_cast<Geom *>(GetTarget())) != nullptr)
103  {
104  geom->GetWorldQuaternion(q);
105  angle = pgd::FindAngle(interpolatedTarget, pgd::Quaternion(q[0], q[1], q[2], q[3]));
106  }
107  else
108  {
109  std::cerr << "DataTargetQuaternion target missing error " << name() << "\n";
110  }
111  return angle;
112 }

167 // this function initialises the data in the object based on the contents
168 // of an xml_node node. It uses information from the simulation as required
169 // to satisfy dependencies
170 // it returns nullptr on success and a pointer to lastError() on failure
171 std::string *DataTargetQuaternion::createFromAttributes()
172 {
173  if (DataTarget::createFromAttributes()) return lastErrorPtr();
174 
175  std::string buf;
176  if (findAttribute("DataType"s, &buf) == nullptr) return lastErrorPtr();
177  m_Target = nullptr;
178  if (findAttribute("TargetID"s, &buf) == nullptr) return lastErrorPtr();
179  for (bool once = true; once; once = false)
180  {
181  auto iterBody = simulation()->GetBodyList()->find(buf);
182  if (iterBody != simulation()->GetBodyList()->end()) { m_Target = iterBody->second.get(); break; }
183  auto iterGeom = simulation()->GetGeomList()->find(buf);
184  if (iterGeom != simulation()->GetGeomList()->end()) { m_Target = iterGeom->second.get(); break; }
185  auto iterMarker = simulation()->GetMarkerList()->find(buf);
186  if (iterMarker != simulation()->GetMarkerList()->end()) { m_Target = iterMarker->second.get(); break; }
187  }
188  if (!m_Target)
189  {
190  setLastError("DataTargetQuaternion ID=\""s + name() +"\" TargetID not found "s + buf);
191  return lastErrorPtr();
192  }
193 
194  if (findAttribute("TargetValues"s, &buf) == nullptr) return lastErrorPtr();
195  std::vector<std::string> targetValuesTokens;
196  pystring::split(buf, targetValuesTokens);
197  if (targetValuesTokens.size() == 0)
198  {
199  setLastError("DataTargetQuaternion ID=\""s + name() +"\" No values found in TargetValues"s);
200  return lastErrorPtr();
201  }
202  if (targetValuesTokens.size() != targetTimeList()->size() * 4)
203  {
204  setLastError("DataTargetQuaternion ID=\""s + name() +"\" Number of values in TargetValues does not match 4 * TargetTimes"s);
205  return lastErrorPtr();
206  }
207  m_QValueList.clear();
208  m_QValueList.reserve(targetTimeList()->size());
209  for (size_t i = 0; i < targetTimeList()->size(); i++)
210  {
211  pgd::Quaternion q(GSUtil::Double(targetValuesTokens[i * 4]), GSUtil::Double(targetValuesTokens[i * 4 + 1]),
212  GSUtil::Double(targetValuesTokens[i * 4 + 2]), GSUtil::Double(targetValuesTokens[i * 4 + 3]));
213  m_QValueList.push_back(q);
214  }
215 
216  if (m_Target) setUpstreamObjects({m_Target});
217  return nullptr;
218 }
