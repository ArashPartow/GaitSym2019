48 std::tuple<double, bool> DataTarget::calculateMatchValue(double time)
49 {
50  switch (m_interpolationType)
51  {
52  case Punctuated:
53  {
54  auto lowerBound = std::lower_bound(m_targetTimeList.begin(), m_targetTimeList.end(), time);
55  size_t index = std::distance(m_targetTimeList.begin(), lowerBound);
56  if (index == 0)
57  return std::make_tuple(m_lastValue, false); // this means that time is less than the lowest value in the list
58  if (index == m_lastIndex)
59  return std::make_tuple(m_lastValue, false);
60  m_lastIndex = index;
61  m_lastValue = m_intercept + m_slope * positiveFunction(calculateError(index - 1));
62  break;
63  }
64  case Continuous:
65  {
66  m_lastValue = m_intercept + m_slope * positiveFunction(calculateError(time));
67  break;
68  }
69  }
70  if (m_lastValue < m_abortBelow || m_lastValue > m_abortAbove)
71  {
72  simulation()->SetDataTargetAbort(name());
73  m_lastValue += m_abortBonus;
74  }
75  return std::make_tuple(m_lastValue, true);
76 }

96 // this function initialises the data in the object based on the contents
97 // of an xml_node node. It uses information from the simulation as required
98 // to satisfy dependencies
99 // it returns nullptr on success and a pointer to lastError() on failure
100 std::string *DataTarget::createFromAttributes()
101 {
102  if (NamedObject::createFromAttributes()) return lastErrorPtr();
103 
104  std::string buf;
105  buf.reserve(10000);
106 
107  if (findAttribute("Intercept"s, &buf) == nullptr) return lastErrorPtr();
108  m_intercept = GSUtil::Double(buf);
109  if (findAttribute("Slope"s, &buf) == nullptr) return lastErrorPtr();
110  m_slope = GSUtil::Double(buf);
111 
112  if (findAttribute("AbortAbove"s, &buf)) m_abortAbove = GSUtil::Double(buf);
113  if (findAttribute("AbortBelow"s, &buf)) m_abortBelow = GSUtil::Double(buf);
114  if (findAttribute("AbortBonus"s, &buf)) m_abortBonus = GSUtil::Double(buf);
115 
116  if (findAttribute("TargetTimes"s, &buf) == nullptr) return lastErrorPtr();
117  std::vector<std::string> targetTimesTokens;
118  pystring::split(buf, targetTimesTokens);
119  if (targetTimesTokens.size() == 0)
120  {
121  setLastError("DataTarget ID=\""s + name() +"\" No times found in TargetTimes"s);
122  return lastErrorPtr();
123  }
124  m_targetTimeList.clear();
125  m_targetTimeList.reserve(targetTimesTokens.size());
126  for (auto &&token : targetTimesTokens) m_targetTimeList.push_back(GSUtil::Double(token));
127  if (std::is_sorted(m_targetTimeList.begin(), m_targetTimeList.end()) == false)
128  {
129  setLastError("DataTarget ID=\""s + name() +"\" TargetTimes are not in ascending order"s);
130  return lastErrorPtr();
131  }
132 
133  if (findAttribute("MatchType"s, &buf) == nullptr) return lastErrorPtr();
134  size_t matchTypeIndex;
135  for (matchTypeIndex = 0; matchTypeIndex < matchTypeCount; matchTypeIndex++)
136  {
137  if (buf == matchTypeStrings(matchTypeIndex))
138  {
139  m_matchType = static_cast<MatchType>(matchTypeIndex);
140  break;
141  }
142  }
143  if (matchTypeIndex >= matchTypeCount)
144  {
145  setLastError("DataTarget ID=\""s + name() +"\" Unrecognised MatchType "s + buf);
146  return lastErrorPtr();
147  }
148 
149  if (findAttribute("InterpolationType"s, &buf) == nullptr) return lastErrorPtr();
150  size_t interpolationTypeIndex;
151  for (interpolationTypeIndex = 0; interpolationTypeIndex < interpolationTypeCount; interpolationTypeIndex++)
152  {
153  if (buf == interpolationTypeStrings(interpolationTypeIndex))
154  {
155  m_interpolationType = static_cast<InterpolationType>(interpolationTypeIndex);
156  break;
157  }
158  }
159  if (interpolationTypeIndex >= interpolationTypeCount)
160  {
161  setLastError("DataTarget ID=\""s + name() +"\" Unrecognised InterpolationType "s + buf);
162  return lastErrorPtr();
163  }
164 
165  return nullptr;
166 }

177 // this function appends data to a pre-existing xml_node - often created by XMLSave
178 void DataTarget::appendToAttributes()
179 {
180  NamedObject::appendToAttributes();
181  std::string buf;
182  buf.reserve(size_t(m_targetTimeList.size()) * 32);
183  setAttribute("Intercept"s, *GSUtil::ToString(m_intercept, &buf));
184  setAttribute("Slope"s, *GSUtil::ToString(m_slope, &buf));
185  setAttribute("AbortAbove"s, *GSUtil::ToString(m_abortAbove, &buf));
186  setAttribute("AbortBelow"s, *GSUtil::ToString(m_abortBelow, &buf));
187  setAttribute("AbortBonus"s, *GSUtil::ToString(m_abortBonus, &buf));
188  setAttribute("TargetTimes"s, *GSUtil::ToString(m_targetTimeList.data(), m_targetTimeList.size(), &buf));
189  setAttribute("MatchType", matchTypeStrings(m_matchType));
190  setAttribute("InterpolationType", interpolationTypeStrings(m_interpolationType));
191 }
