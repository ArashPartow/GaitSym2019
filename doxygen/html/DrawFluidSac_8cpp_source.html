47 void DrawFluidSac::initialise(SimulationWidget *simulationWidget)
48 {
49  if (!m_fluidSac) return;
50 
51  m_fluidSacColour.setRedF(qreal(m_fluidSac->colour1().r()));
52  m_fluidSacColour.setGreenF(qreal(m_fluidSac->colour1().g()));
53  m_fluidSacColour.setBlueF(qreal(m_fluidSac->colour1().b()));
54  m_fluidSacColour.setAlphaF(qreal(m_fluidSac->colour1().alpha()));
55  m_fluidSacForceColour.setRedF(qreal(m_fluidSac->colour2().r()));
56  m_fluidSacForceColour.setGreenF(qreal(m_fluidSac->colour2().g()));
57  m_fluidSacForceColour.setBlueF(qreal(m_fluidSac->colour2().b()));
58  m_fluidSacForceColour.setAlphaF(qreal(m_fluidSac->colour2().alpha()));
59 
60  m_fluidSacForceRadius = m_fluidSac->size1();
61  m_fluidSacForceScale = m_fluidSac->size2();
62 
63  m_facetedObject = std::make_unique<FacetedObject>();
64  m_facetedObject->setSimulationWidget(simulationWidget);
65  m_facetedObject->setBlendColour(m_fluidSacColour, 1);
66  m_facetedObject->AllocateMemory(m_fluidSac->numTriangles());
67  double vertices[9];
68  for (size_t i = 0; i < m_fluidSac->numTriangles(); i++)
69  {
70  m_fluidSac->triangleVertices(i, vertices);
71  m_facetedObject->AddTriangle(vertices);
72  }
73 // qDebug() << "DrawFluidSac " << facetedObject->GetNumTriangles() << " triangles created\n";
74 
75  if (m_displayFluidSacForces)
76  {
77  const std::vector<PointForce> &pointForceList = m_fluidSac->pointForceList();
78  for (size_t i = 0; i < pointForceList.size(); i++)
79  {
80  std::vector<pgd::Vector3> polyline;
81  polyline.clear();
82  pgd::Vector3 f = pgd::Vector3(pointForceList.at(i).vector[0], pointForceList.at(i).vector[1], pointForceList.at(i).vector[2]) * m_fluidSacForceScale;
83  polyline.push_back(pgd::Vector3(pointForceList.at(i).point[0], pointForceList.at(i).point[1], pointForceList.at(i).point[2]));
84  polyline.push_back(pgd::Vector3(pointForceList.at(i).point[0], pointForceList.at(i).point[1], pointForceList.at(i).point[2]) + f);
85  std::unique_ptr<FacetedObject> facetedPolyline = std::make_unique<FacetedPolyline>(&polyline, m_fluidSacForceRadius, m_fluidSacForceSegments, m_fluidSacForceColour, 1);
86  m_facetedObjectForceList.push_back(std::move(facetedPolyline));
87  }
88  }
89  m_facetedObjectList.push_back(m_facetedObject.get());
90 }
