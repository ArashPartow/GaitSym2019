68 void DrawJoint::initialise(SimulationWidget *simulationWidget)
69 {
70  if (!m_joint) return;
71 
72  m_jointColor.setRedF(qreal(m_joint->colour1().r()));
73  m_jointColor.setGreenF(qreal(m_joint->colour1().g()));
74  m_jointColor.setBlueF(qreal(m_joint->colour1().b()));
75  m_jointColor.setAlphaF(qreal(m_joint->colour1().alpha()));
76  m_jointAxisSize = m_joint->size1();
77 
78  HingeJoint *hingeJoint = dynamic_cast<HingeJoint *>(m_joint);
79  if (hingeJoint)
80  {
81  double halfLen = m_jointAxisSize / 2.0;
82  std::vector<pgd::Vector3> polyLine;
83  polyLine.reserve(2);
84  polyLine.push_back(pgd::Vector3(-halfLen, 0, 0));
85  polyLine.push_back(pgd::Vector3(+halfLen, 0, 0));
86  m_facetedObject1 = std::make_unique<FacetedPolyline>(&polyLine, halfLen / 10, m_jointSegments, m_jointColor, 1);
87  m_facetedObject1->setSimulationWidget(simulationWidget);
88  m_facetedObjectList.push_back(m_facetedObject1.get());
89  return;
90  }
91 
92  UniversalJoint *universalJoint = dynamic_cast<UniversalJoint *>(m_joint);
93  if (universalJoint)
94  {
95  double halfLen = m_jointAxisSize / 2.0;
96  std::vector<pgd::Vector3> polyLine;
97  polyLine.reserve(2);
98  polyLine.push_back(pgd::Vector3(-halfLen, 0, 0));
99  polyLine.push_back(pgd::Vector3(+halfLen, 0, 0));
100  m_facetedObject1 = std::make_unique<FacetedPolyline>(&polyLine, halfLen / 10, m_jointSegments, m_jointColor, 1);
101  polyLine[0] = pgd::Vector3(0, -halfLen, 0);
102  polyLine[1] = pgd::Vector3(0, +halfLen, 0);
103  m_facetedObject2 = std::make_unique<FacetedPolyline>(&polyLine, halfLen / 10, m_jointSegments, m_jointColor, 1);
104  m_facetedObject1->setSimulationWidget(simulationWidget);
105  m_facetedObject2->setSimulationWidget(simulationWidget);
106  m_facetedObjectList.push_back(m_facetedObject1.get());
107  m_facetedObjectList.push_back(m_facetedObject2.get());
108  return;
109  }
110 
111  BallJoint *ballJoint = dynamic_cast<BallJoint *>(m_joint);
112  if (ballJoint)
113  {
114  double halfLen = m_jointAxisSize / 2.0;
115  std::vector<pgd::Vector3> polyLine;
116  polyLine.reserve(2);
117  polyLine.push_back(pgd::Vector3(-halfLen, 0, 0));
118  polyLine.push_back(pgd::Vector3(+halfLen, 0, 0));
119  m_facetedObject1 = std::make_unique<FacetedPolyline>(&polyLine, halfLen / 10, m_jointSegments, m_jointColor, 1);
120  polyLine[0] = pgd::Vector3(0, -halfLen, 0);
121  polyLine[1] = pgd::Vector3(0, +halfLen, 0);
122  m_facetedObject2 = std::make_unique<FacetedPolyline>(&polyLine, halfLen / 10, m_jointSegments, m_jointColor, 1);
123  polyLine[0] = pgd::Vector3(0, 0, -halfLen);
124  polyLine[1] = pgd::Vector3(0, 0, +halfLen);
125  m_facetedObject3 = std::make_unique<FacetedPolyline>(&polyLine, halfLen / 10, m_jointSegments, m_jointColor, 1);
126  m_facetedObject1->setSimulationWidget(simulationWidget);
127  m_facetedObject2->setSimulationWidget(simulationWidget);
128  m_facetedObject3->setSimulationWidget(simulationWidget);
129  m_facetedObjectList.push_back(m_facetedObject1.get());
130  m_facetedObjectList.push_back(m_facetedObject2.get());
131  m_facetedObjectList.push_back(m_facetedObject3.get());
132  return;
133  }
134 
135  FixedJoint *fixedJoint = dynamic_cast<FixedJoint *>(m_joint);
136  if (fixedJoint && fixedJoint->GetStressCalculationType() != FixedJoint::none)
137  {
138  qDebug() << "Debug DrawJoint::initialise:" << m_joint->name().c_str();
139  m_facetedObject1 = std::make_unique<FacetedRect>(fixedJoint->width(), fixedJoint->height(), m_jointColor, 1);
140  m_facetedObject1->setSimulationWidget(simulationWidget);
141  m_facetedObject1->Move((fixedJoint->width() / 2) - fixedJoint->xOrigin(), (fixedJoint->height() / 2) - fixedJoint->yOrigin(), 0);
142  fixedJoint->CalculatePixmap();
143  std::unique_ptr<QOpenGLTexture> texture = std::make_unique<QOpenGLTexture>(QOpenGLTexture::Target2D);
144  texture->setAutoMipMapGenerationEnabled(false);
145  texture->setFormat(QOpenGLTexture::RGBA8_UNorm); // this maps to QImage::Format_RGBA8888
146  texture->setSize(int(fixedJoint->nx()), int(fixedJoint->ny()), 1);
147  texture->setMipLevels(1);
148  texture->allocateStorage(QOpenGLTexture::RGBA, QOpenGLTexture::UInt8);
149  QOpenGLPixelTransferOptions uploadOptions;
150  uploadOptions.setAlignment(1);
151  texture->setData(0, QOpenGLTexture::RGBA, QOpenGLTexture::UInt8, fixedJoint->pixMap().data(), &uploadOptions);
152  texture->setMinificationFilter(QOpenGLTexture::Nearest);
153  texture->setMagnificationFilter(QOpenGLTexture::Nearest);
154  texture->setWrapMode(QOpenGLTexture::ClampToEdge);
155  m_facetedObject1->setTexture(std::move(texture));
156  m_facetedObject1->setDecal(1);
157  m_facetedObjectList.push_back(m_facetedObject1.get());
158  return;
159  }

168  pgd::Vector3 p = marker->GetWorldPosition();
