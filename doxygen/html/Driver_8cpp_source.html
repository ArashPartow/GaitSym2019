82 // this function initialises the data in the object based on the contents
83 // of an xml_node node. It uses information from the simulation as required
84 // to satisfy dependencies
85 // it returns nullptr on success and a pointer to lastError() on failure
86 std::string *Driver::createFromAttributes()
87 {
88  if (NamedObject::createFromAttributes()) return lastErrorPtr();
89 
90  std::string buf;
91  double doubleList[2];
92 
93  if (findAttribute("TargetIDList"s, &buf) == nullptr) return lastErrorPtr();
94  std::vector<std::string> targetNames;
95  pystring::split(buf, targetNames);
96 // if (targetNames.size() == 0)
97 // {
98 // setLastError("Driver ID=\""s + name() +"\" TargetIDList is empty"s);
99 // return lastErrorPtr();
100 // }
101  m_targetList.clear();
102  std::vector<NamedObject *> upstreamObjects;
103  upstreamObjects.reserve(targetNames.size());
104  for (size_t i = 0; i < targetNames.size(); i++)
105  {
106  auto muscleIter = simulation()->GetMuscleList()->find(targetNames[i]);
107  if (muscleIter != simulation()->GetMuscleList()->end())
108  {
109  this->AddTarget(muscleIter->second.get());
110  upstreamObjects.push_back(muscleIter->second.get());
111  }
112  else
113  {
114  auto controllerIter = simulation()->GetControllerList()->find(targetNames[i]);
115  if (controllerIter != simulation()->GetControllerList()->end())
116  {
117  this->AddTarget(controllerIter->second.get());
118  upstreamObjects.push_back(controllerIter->second.get());
119  }
120  else
121  {
122  setLastError("Driver ID=\""s + name() +"\" TargetID=\""s + buf + "\" not found"s);
123  return lastErrorPtr();
124  }
125  }
126  }
127 
128  if (findAttribute("DriverRange"s, &buf))
129  {
130  GSUtil::Double(buf, 2, doubleList);
131  this->setMinValue(doubleList[0]);
132  this->setMaxValue(doubleList[1]);
133  }
134 
135  if (findAttribute("LinearInterpolation"s, &buf))
136  {
137  this->setInterp(GSUtil::Bool(buf));
138  }
139 
140  setUpstreamObjects(std::move(upstreamObjects));
141  return nullptr;
142 }

153 // this function appends data to a pre-existing xml_node - often created by XMLSave
154 void Driver::appendToAttributes()
155 {
156  NamedObject::appendToAttributes();
157  std::string buf;
158  std::vector<std::string> stringList;
159  stringList.reserve(m_targetList.size());
160  for (auto it = m_targetList.begin(); it != m_targetList.end(); it++) stringList.push_back(it->first);
161  setAttribute("TargetIDList"s, pystring::join(" "s, stringList));
162  double doubleList[2] = { m_minValue, m_maxValue };
163  setAttribute("DriverRange"s, *GSUtil::ToString(doubleList, 2, &buf));
164  setAttribute("LinearInterpolation"s, *GSUtil::ToString(Interp(), &buf));
165 }
