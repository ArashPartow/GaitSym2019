121 /*
122  * To speed up computations, we have tables for logarithm, exponent
123  * and inverse of a number. If GF_BITS <= 8, we use a table for
124  * multiplication as well (it takes 64K, no big deal even on a PDA,
125  * especially because it can be pre-initialized and put into a ROM!),
126  * otherwhise we use a table of logarithms.
127  * In any case the macro gf_mul(x,y) takes care of multiplications.
128  */
129 
130 static gf gf_exp[2*GF_SIZE]; /* index->poly form conversion table */
131 static int gf_log[GF_SIZE + 1]; /* Poly->index form conversion table */
132 static gf inverse[GF_SIZE+1]; /* inverse of field elem. */
133 /* inv[\alpha**i]=\alpha**(GF_SIZE-i-1) */
134 
135 /*
136  * modnn(x) computes x % GF_SIZE, where GF_SIZE is 2**GF_BITS - 1,
137  * without a slow divide.
138  */
139 static inline gf
140 modnn(int x)
141 {
142  while (x >= GF_SIZE) {
143  x -= GF_SIZE;
144  x = (x >> GF_BITS) + (x & GF_SIZE);
145  }
146  return x;
147 }

149 #define SWAP(a,b,t) {t tmp; tmp=a; a=b; b=tmp;}

151 /*
152  * gf_mul(x,y) multiplies two numbers. If GF_BITS<=8, it is much
153  * faster to use a multiplication table.
154  *
155  * USE_GF_MULC, GF_MULC0(c) and GF_ADDMULC(x) can be used when multiplying
156  * many numbers by the same constant. In this case the first
157  * call sets the constant, and others perform the multiplications.
158  * A value related to the multiplication is held in a local variable
159  * declared with USE_GF_MULC . See usage in addmul1().
160  */
161 #if (GF_BITS <= 8)
162 static gf gf_mul_table[GF_SIZE + 1][GF_SIZE + 1];
163 
164 #define gf_mul(x,y) gf_mul_table[x][y]
165 
166 #define USE_GF_MULC /* register */ gf * __gf_mulc_
167 #define GF_MULC0(c) __gf_mulc_ = gf_mul_table[c]
168 #define GF_ADDMULC(dst, x) dst ^= __gf_mulc_[x]
169 
170 static void
171 init_mul_table()
172 {
173  int i, j;
174  for (i=0; i< GF_SIZE+1; i++)
175  for (j=0; j< GF_SIZE+1; j++)
176  gf_mul_table[i][j] = gf_exp[modnn(gf_log[i] + gf_log[j]) ] ;
177 
178  for (j=0; j< GF_SIZE+1; j++)
179  gf_mul_table[0][j] = gf_mul_table[j][0] = 0;
180 }
181 #else /* GF_BITS > 8 */
182 static inline gf
183 gf_mul(gf x, gf y)
184 {
185  if ( (x) == 0 || (y)==0 ) return 0;
186 
187  return gf_exp[gf_log[x] + gf_log[y] ] ;
188 }
189 #define init_mul_table()
190 
191 #define USE_GF_MULC /* register */ gf * __gf_mulc_
192 #define GF_MULC0(c) __gf_mulc_ = &gf_exp[ gf_log[c] ]
193 #define GF_ADDMULC(dst, x) { if (x) dst ^= __gf_mulc_[ gf_log[x] ] ; }
194 #endif

196 /*
197  * Generate GF(2**m) from the irreducible polynomial p(X) in p[0]..p[m]
198  * Lookup tables:
199  * index->polynomial form gf_exp[] contains j= \alpha^i;
200  * polynomial form -> index form gf_log[ j = \alpha^i ] = i
201  * \alpha=x is the primitive element of GF(2^m)
202  *
203  * For efficiency, gf_exp[] has size 2*GF_SIZE, so that a simple
204  * multiplication of two numbers can be resolved without calling modnn
205  */
206 
207 #define NEW_GF_MATRIX(rows, cols) \
208  (gf *)malloc((rows) * (cols) * sizeof(gf))
209 
210 /*
211  * initialize the data structures used for computations in GF.
212  */
213 static void
214 generate_gf(void)
215 {
216  int i;
217  gf mask;
218  const char *Pp = allPp[GF_BITS] ;
219 
220  mask = 1; /* x ** 0 = 1 */
221  gf_exp[GF_BITS] = 0; /* will be updated at the end of the 1st loop */
222  /*
223  * first, generate the (polynomial representation of) powers of \alpha,
224  * which are stored in gf_exp[i] = \alpha ** i .
225  * At the same time build gf_log[gf_exp[i]] = i .
226  * The first GF_BITS powers are simply bits shifted to the left.
227  */
228  for (i = 0; i < GF_BITS; i++, mask <<= 1 ) {
229  gf_exp[i] = mask;
230  gf_log[gf_exp[i]] = i;
231  /*
232  * If Pp[i] == 1 then \alpha ** i occurs in poly-repr
233  * gf_exp[GF_BITS] = \alpha ** GF_BITS
234  */
235  if ( Pp[i] == '1' )
236  gf_exp[GF_BITS] ^= mask;
237  }
238  /*
239  * now gf_exp[GF_BITS] = \alpha ** GF_BITS is complete, so can als
240  * compute its inverse.
241  */
242  gf_log[gf_exp[GF_BITS]] = GF_BITS;
243  /*
244  * Poly-repr of \alpha ** (i+1) is given by poly-repr of
245  * \alpha ** i shifted left one-bit and accounting for any
246  * \alpha ** GF_BITS term that may occur when poly-repr of
247  * \alpha ** i is shifted.
248  */
249  mask = 1 << (GF_BITS - 1 ) ;
250  for (i = GF_BITS + 1; i < GF_SIZE; i++) {
251  if (gf_exp[i - 1] >= mask)
252  gf_exp[i] = gf_exp[GF_BITS] ^ ((gf_exp[i - 1] ^ mask) << 1);
253  else
254  gf_exp[i] = gf_exp[i - 1] << 1;
255  gf_log[gf_exp[i]] = i;
256  }
257  /*
258  * log(0) is not defined, so use a special value
259  */
260  gf_log[0] = GF_SIZE ;
261  /* set the extended gf_exp values for fast multiply */
262  for (i = 0 ; i < GF_SIZE ; i++)
263  gf_exp[i + GF_SIZE] = gf_exp[i] ;
264 
265  /*
266  * again special cases. 0 has no inverse. This used to
267  * be initialized to GF_SIZE, but it should make no difference
268  * since noone is supposed to read from here.
269  */
270  inverse[0] = 0 ;
271  inverse[1] = 1;
272  for (i=2; i<=GF_SIZE; i++)
273  inverse[i] = gf_exp[GF_SIZE-gf_log[i]];
274 }

292 #define UNROLL 16 /* 1, 4, 8, 16 */
293 static void
294 addmul1(gf *dst1, gf *src1, gf c, int sz)
295 {
296  USE_GF_MULC ;
297  /* register */ gf *dst = dst1, *src = src1 ;
298  gf *lim = &dst[sz - UNROLL + 1] ;
299 
300  GF_MULC0(c) ;
301 
302 #if (UNROLL > 1) /* unrolling by 8/16 is quite effective on the pentium */
303  for (; dst < lim ; dst += UNROLL, src += UNROLL ) {
304  GF_ADDMULC( dst[0] , src[0] );
305  GF_ADDMULC( dst[1] , src[1] );
306  GF_ADDMULC( dst[2] , src[2] );
307  GF_ADDMULC( dst[3] , src[3] );
308 #if (UNROLL > 4)
309  GF_ADDMULC( dst[4] , src[4] );
310  GF_ADDMULC( dst[5] , src[5] );
311  GF_ADDMULC( dst[6] , src[6] );
312  GF_ADDMULC( dst[7] , src[7] );
313 #endif
314 #if (UNROLL > 8)
315  GF_ADDMULC( dst[8] , src[8] );
316  GF_ADDMULC( dst[9] , src[9] );
317  GF_ADDMULC( dst[10] , src[10] );
318  GF_ADDMULC( dst[11] , src[11] );
319  GF_ADDMULC( dst[12] , src[12] );
320  GF_ADDMULC( dst[13] , src[13] );
321  GF_ADDMULC( dst[14] , src[14] );
322  GF_ADDMULC( dst[15] , src[15] );
323 #endif
324  }
325 #endif
326  lim += UNROLL - 1 ;
327  for (; dst < lim; dst++, src++ ) /* final components */
328  GF_ADDMULC( *dst , *src );
329 }

331 /*
332  * computes C = AB where A is n*k, B is k*m, C is n*m
333  */
334 static void
335 matmul(gf *a, gf *b, gf *c, int n, int k, int m)
336 {
337  int row, col, i ;
338 
339  for (row = 0; row < n ; row++) {
340  for (col = 0; col < m ; col++) {
341  gf *pa = &a[ row * k ];
342  gf *pb = &b[ col ];
343  gf acc = 0 ;
344  for (i = 0; i < k ; i++, pa++, pb += m )
345  acc ^= gf_mul( *pa, *pb ) ;
346  c[ row * m + col ] = acc ;
347  }
348  }
349 }

371 /*
372  * invert_mat() takes a matrix and produces its inverse
373  * k is the size of the matrix.
374  * (Gauss-Jordan, adapted from Numerical Recipes in C)
375  * Return non-zero if singular.
376  */
377 DEB( int pivloops=0; int pivswaps=0 ; /* diagnostic */)
378 static int
379 invert_mat(gf *src, int k)
380 {
381  gf c, *p ;
382  int irow, icol, row, col, i, ix ;
383 
384  int error = 1 ;
385  int *indxc = (int *)malloc(k*sizeof(int));
386  int *indxr = (int *)malloc(k*sizeof(int));
387  int *ipiv = (int *)malloc(k*sizeof(int));
388  gf *id_row = NEW_GF_MATRIX(1, k);
389  gf *temp_row = NEW_GF_MATRIX(1, k);
390 
391  bzero(id_row, k*sizeof(gf));
392  DEB( pivloops=0; pivswaps=0 ; /* diagnostic */ )
393  /*
394  * ipiv marks elements already used as pivots.
395  */
396  for (i = 0; i < k ; i++)
397  ipiv[i] = 0 ;
398 
399  for (col = 0; col < k ; col++) {
400  gf *pivot_row ;
401  /*
402  * Zeroing column 'col', look for a non-zero element.
403  * First try on the diagonal, if it fails, look elsewhere.
404  */
405  irow = icol = -1 ;
406  if (ipiv[col] != 1 && src[col*k + col] != 0) {
407  irow = col ;
408  icol = col ;
409  goto found_piv ;
410  }
411  for (row = 0 ; row < k ; row++) {
412  if (ipiv[row] != 1) {
413  for (ix = 0 ; ix < k ; ix++) {
414  DEB( pivloops++ ; )
415  if (ipiv[ix] == 0) {
416  if (src[row*k + ix] != 0) {
417  irow = row ;
418  icol = ix ;
419  goto found_piv ;
420  }
421  } else if (ipiv[ix] > 1) {
422  fprintf(stderr, "singular matrix\n");
423  goto fail ;
424  }
425  }
426  }
427  }
428  if (icol == -1) {
429  fprintf(stderr, "XXX pivot not found!\n");
430  goto fail ;
431  }
432 found_piv:
433  ++(ipiv[icol]) ;
434  /*
435  * swap rows irow and icol, so afterwards the diagonal
436  * element will be correct. Rarely done, not worth
437  * optimizing.
438  */
439  if (irow != icol) {
440  for (ix = 0 ; ix < k ; ix++ ) {
441  SWAP( src[irow*k + ix], src[icol*k + ix], gf) ;
442  }
443  }
444  indxr[col] = irow ;
445  indxc[col] = icol ;
446  pivot_row = &src[icol*k] ;
447  c = pivot_row[icol] ;
448  if (c == 0) {
449  fprintf(stderr, "singular matrix 2\n");
450  goto fail ;
451  }
452  if (c != 1 ) { /* otherwhise this is a NOP */
453  /*
454  * this is done often , but optimizing is not so
455  * fruitful, at least in the obvious ways (unrolling)
456  */
457  DEB( pivswaps++ ; )
458  c = inverse[ c ] ;
459  pivot_row[icol] = 1 ;
460  for (ix = 0 ; ix < k ; ix++ )
461  pivot_row[ix] = gf_mul(c, pivot_row[ix] );
462  }
463  /*
464  * from all rows, remove multiples of the selected row
465  * to zero the relevant entry (in fact, the entry is not zero
466  * because we know it must be zero).
467  * (Here, if we know that the pivot_row is the identity,
468  * we can optimize the addmul).
469  */
470  id_row[icol] = 1;
471  if (bcmp(pivot_row, id_row, k*sizeof(gf)) != 0) {
472  for (p = src, ix = 0 ; ix < k ; ix++, p += k ) {
473  if (ix != icol) {
474  c = p[icol] ;
475  p[icol] = 0 ;
476  addmul(p, pivot_row, c, k );
477  }
478  }
479  }
480  id_row[icol] = 0;
481  } /* done all columns */
482  for (col = k-1 ; col >= 0 ; col-- ) {
483  if (indxr[col] <0 || indxr[col] >= k)
484  fprintf(stderr, "AARGH, indxr[col] %d\n", indxr[col]);
485  else if (indxc[col] <0 || indxc[col] >= k)
486  fprintf(stderr, "AARGH, indxc[col] %d\n", indxc[col]);
487  else
488  if (indxr[col] != indxc[col] ) {
489  for (row = 0 ; row < k ; row++ ) {
490  SWAP( src[row*k + indxr[col]], src[row*k + indxc[col]], gf) ;
491  }
492  }
493  }
494  error = 0 ;
495 fail:
496  free(indxc);
497  free(indxr);
498  free(ipiv);
499  free(id_row);
500  free(temp_row);
501  return error ;
502 }

504 /*
505  * fast code for inverting a vandermonde matrix.
506  * XXX NOTE: It assumes that the matrix
507  * is not singular and _IS_ a vandermonde matrix. Only uses
508  * the second column of the matrix, containing the p_i's.
509  *
510  * Algorithm borrowed from "Numerical recipes in C" -- sec.2.8, but
511  * largely revised for my purposes.
512  * p = coefficients of the matrix (p_i)
513  * q = values of the polynomial (known)
514  */
515 
516 int
517 invert_vdm(gf *src, int k)
518 {
519  int i, j, row, col ;
520  gf *b, *c, *p;
521  gf t, xx ;
522 
523  if (k == 1) /* degenerate case, matrix must be p^0 = 1 */
524  return 0 ;
525  /*
526  * c holds the coefficient of P(x) = Prod (x - p_i), i=0..k-1
527  * b holds the coefficient for the matrix inversion
528  */
529  c = NEW_GF_MATRIX(1, k);
530  b = NEW_GF_MATRIX(1, k);
531 
532  p = NEW_GF_MATRIX(1, k);
533 
534  for ( j=1, i = 0 ; i < k ; i++, j+=k ) {
535  c[i] = 0 ;
536  p[i] = src[j] ; /* p[i] */
537  }
538  /*
539  * construct coeffs. recursively. We know c[k] = 1 (implicit)
540  * and start P_0 = x - p_0, then at each stage multiply by
541  * x - p_i generating P_i = x P_{i-1} - p_i P_{i-1}
542  * After k steps we are done.
543  */
544  c[k-1] = p[0] ; /* really -p(0), but x = -x in GF(2^m) */
545  for (i = 1 ; i < k ; i++ ) {
546  gf p_i = p[i] ; /* see above comment */
547  for (j = k-1 - ( i - 1 ) ; j < k-1 ; j++ )
548  c[j] ^= gf_mul( p_i, c[j+1] ) ;
549  c[k-1] ^= p_i ;
550  }
551 
552  for (row = 0 ; row < k ; row++ ) {
553  /*
554  * synthetic division etc.
555  */
556  xx = p[row] ;
557  t = 1 ;
558  b[k-1] = 1 ; /* this is in fact c[k] */
559  for (i = k-2 ; i >= 0 ; i-- ) {
560  b[i] = c[i+1] ^ gf_mul(xx, b[i+1]) ;
561  t = gf_mul(xx, t) ^ b[i] ;
562  }
563  for (col = 0 ; col < k ; col++ )
564  src[col*k + row] = gf_mul(inverse[t], b[col] );
565  }
566  free(c) ;
567  free(b) ;
568  free(p) ;
569  return 0 ;
570 }

609  * the encoding matrix.
610  */
611 struct fec_parms *
612  fec_new(int k, int n)
613 {
614  int row, col ;
615  gf *p, *tmp_m ;
616 
617  struct fec_parms *retval ;
618 
619  if (fec_initialized == 0)
620  init_fec();
621 
622  if (k > GF_SIZE + 1 || n > GF_SIZE + 1 || k > n ) {
623  fprintf(stderr, "Invalid parameters k %d n %d GF_SIZE %d\n",
624  k, n, GF_SIZE );

626  }
627  retval = (struct fec_parms *)malloc(sizeof(struct fec_parms));
628  retval->k = k ;
629  retval->n = n ;
630  retval->enc_matrix = NEW_GF_MATRIX(n, k);
631  retval->magic = ( ( FEC_MAGIC ^ k) ^ n) ^ (int)((intptr_t)retval->enc_matrix) ;

633  /*
634  * fill the matrix with powers of field elements, starting from 0.
635  * The first row is special, cannot be computed with exp. table.
636  */
637  tmp_m[0] = 1 ;
638  for (col = 1; col < k ; col++)
639  tmp_m[col] = 0 ;
640  for (p = tmp_m + k, row = 0; row < n-1 ; row++, p += k) {
641  for ( col = 0 ; col < k ; col ++ )
642  p[col] = gf_exp[modnn(row*col)];
643  }
644 
645  /*
646  * quick code to build systematic matrix: invert the top
647  * k*k vandermonde matrix, multiply right the bottom n-k rows
648  * by the inverse, and construct the identity matrix at the top.
649  */
650  TICK(ticks[3]);
651  invert_vdm(tmp_m, k); /* much faster than invert_mat */
652  matmul(tmp_m + k*k, tmp_m, retval->enc_matrix + k*k, n - k, k, k);
653  /*
654  * the upper matrix is I so do not bother with a slow multiply
655  */
656  bzero(retval->enc_matrix, k*k*sizeof(gf) );
657  for (p = retval->enc_matrix, col = 0 ; col < k ; col++, p += k+1 )
658  *p = 1 ;
659  free(tmp_m);
660  TOCK(ticks[3]);
661 
662  DDB(fprintf(stderr, "--- %ld us to build encoding matrix\n",
663  ticks[3]);)
664  DEB(pr_matrix(retval->enc_matrix, n, k, "encoding_matrix");)
665  return retval ;
666 }

769 /*
770  * fec_decode receives as input a vector of packets, the indexes of
771  * packets, and produces the correct vector as output.
772  *
773  * Input:
774  * code: pointer to code descriptor
775  * pkt: pointers to received packets. They are modified
776  * to store the output packets (in place)
777  * index: pointer to packet indexes (modified)
778  * sz: size of each packet
779  */
780 int
781 fec_decode(struct fec_parms *code, gf *pkt[], int index[], int sz)
782 {
783  gf *m_dec ;
784  gf **new_pkt ;
785  int row, col , k = code->k ;
786 
787  if (GF_BITS > 8)
788  sz /= 2 ;
789 
790  if (shuffle(pkt, index, k)) /* error if true */
791  return 1 ;
792  m_dec = build_decode_matrix(code, pkt, index);
793 
794  if (m_dec == NULL)
795  return 1 ; /* error */
796  /*
797  * do the actual decoding
798  */
799  new_pkt = (gf **)malloc (k * sizeof (gf * ));
800  for (row = 0 ; row < k ; row++ ) {
801  if (index[row] >= k) {
802  new_pkt[row] = (gf *)malloc (sz * sizeof (gf));
803  bzero(new_pkt[row], sz * sizeof(gf) ) ;
804  for (col = 0 ; col < k ; col++ )
805  addmul(new_pkt[row], pkt[col], m_dec[row*k + col], sz) ;
806  }
807  }
808  /*
809  * move pkts to their final destination
810  */
811  for (row = 0 ; row < k ; row++ ) {
812  if (index[row] >= k) {
813  bcopy(new_pkt[row], pkt[row], sz*sizeof(gf));
814  free(new_pkt[row]);
815  }
816  }
817  free(new_pkt);
818  free(m_dec);
819 
820  return 0;
821 }
