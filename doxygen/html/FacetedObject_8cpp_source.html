71 // parse an OBJ file to a FacetedObject
72 // returns 0 on success
73 // this tries to be quite fast and not very flexible

76 {
77  m_filename = filename;
78  MeshStoreObject *meshStoreObject = m_meshStore.getMesh(filename);
79  if (meshStoreObject)

94  // read the whole file into memory
95  DataFile theFile;

99

107  {
108  size_t vertex[3];

110  size_t uv[3];
111  OBJMaterial *material;
112  };
113  Triangle triangle;
114  std::vector<Triangle> triangleList;
115  // having checked a few files ball park is 20M file, 100k vertices, 200k faces, 200k normals
116  size_t estimated_max_capacity = theFile.GetSize() / 25; // this will give a factor of 4 to 8 leeway
117  vertexList.reserve(estimated_max_capacity);
118  normalList.reserve(estimated_max_capacity);

125  std::map<std::string, OBJMaterial> materialMap;
126  OBJMaterial *currentMaterial = nullptr;
127  std::string line;
128  line.reserve(1024);
129  while (ptr < endPtr)
130  {
131  // non content (whitespace)
132  if (*ptr < 32)
133  {
134  while (*ptr < 32 && ptr < endPtr) { ptr++; }
135  continue;
136  }
137 
138  // comments
139  if (*ptr == '#')
140  {
141  ptr++;
142  while (*ptr != '\n' && *ptr != '\r' && ptr < endPtr) { ptr++; }
143  ptr++;
144  continue;
145  }
146 
147  // materials definitions
148  if (*ptr == 'm')
149  {
150  // first extract the line
151  line.clear();
152  while (*ptr != '\n' && *ptr != '\r' && ptr < endPtr)
153  {
154  line.append(1, *ptr);
155  ptr++;
156  }
157  if (pystring::startswith(line, "mtllib "s))
158  {

160  if (ParseOBJMaterialFile(materialsFile, &materialMap))
161  {
162  qDebug() << "Error reading material file \"" << materialsFile.c_str() << "\"";
163  }
164  }
165  continue;
166  }
167 
168  // materials use
169  if (*ptr == 'u')
170  {
171  // first extract the line
172  line.clear();
173  while (*ptr != '\n' && *ptr != '\r' && ptr < endPtr)
174  {
175  line.append(1, *ptr);
176  ptr++;
177  }

179  {

181  auto it = materialMap.find(material);
182  if (it == materialMap.end()) currentMaterial = nullptr;

184  }
185  continue;
186  }
187 
188  // vertices and normals and texture coords
189  if (*ptr == 'v')
190  {
191  ptr++;
192  if (ptr >= endPtr) continue;
193  if (*ptr == ' ')
194  {
195  vertex.x = GSUtil::fast_a_to_double(ptr, &ptr);
196  if (ptr >= endPtr) continue;
197  vertex.y = GSUtil::fast_a_to_double(ptr, &ptr);

201  if (vertex.x < m_lowerBound[0]) m_lowerBound[0] = vertex.x;
202  if (vertex.y < m_lowerBound[1]) m_lowerBound[1] = vertex.y;
203  if (vertex.z < m_lowerBound[2]) m_lowerBound[2] = vertex.z;

205  if (vertex.y > m_upperBound[1]) m_upperBound[1] = vertex.y;
206  if (vertex.z > m_upperBound[2]) m_upperBound[2] = vertex.z;
207  ptr++;
208  continue;
209  }
210  if (*ptr == 'n')
211  {
212  ptr++;
213  if (ptr >= endPtr) continue;
214  normal.x = GSUtil::fast_a_to_double(ptr, &ptr);
215  if (ptr >= endPtr) continue;

217  if (ptr >= endPtr) continue;
218  normal.z = GSUtil::fast_a_to_double(ptr, &ptr);
219  normalList.push_back(normal);
220  ptr++;
221  continue;
222  }

224  {
225  ptr++;
226  if (ptr >= endPtr) continue;
227  uv.x = GSUtil::fast_a_to_double(ptr, &ptr);
228  if (ptr >= endPtr) continue;
229  uv.y = GSUtil::fast_a_to_double(ptr, &ptr);
230  uvList.push_back(uv);
231  ptr++;
232  continue;
233  }
234  // not recognised so skip to the end of line
235  while (*ptr != '\n' && *ptr != '\r' && ptr < endPtr) { ptr++; }
236  ptr++;
237  continue;
238  }
239 
240  // faces (only triangular faces supported)
241  if (*ptr == 'f')
242  {
243  ptr++;
244  while (faceFormat == unknown)
245  {
246  const std::regex vertex_only_tester("[ \t]*[0-9]+[ \t]+[0-9]+[ \t]+[0-9]+[ \t]*"s);
247  const std::regex vertex_texture_tester("[ \t]*[0-9]+/[0-9]+[ \t]+[0-9]+/[0-9]+[ \t]+[0-9]+/[0-9]+[ \t]*"s);

250  const char *localEndPtr = ptr + 1;
251  while (*localEndPtr != '\0' && *localEndPtr != '\r' && *localEndPtr != '\n' && localEndPtr < endPtr) localEndPtr++;

255  if (std::regex_match(testStr, sm, vertex_texture_tester)) { faceFormat = vertex_texture; break; }
256  if (std::regex_match(testStr, sm, vertex_normal_tester)) { faceFormat = vertex_normal; break; }
257  if (std::regex_match(testStr, sm, vertex_texture_normal_tester)) { faceFormat = vertex_texture_normal; break; }
258  break;
259  }

262  {

264  if (ptr >= endPtr) continue;
265  triangle.vertex[1] = GSUtil::fast_a_to_uint64_t(ptr, &ptr) - 1;

275  ptr++;

277  {
278 
279  std::swap(triangle.vertex[1], triangle.vertex[2]);
280  std::swap(triangle.normal[1], triangle.normal[2]);
281  std::swap(triangle.uv[1], triangle.uv[2]);
282  triangleList.push_back(triangle);
283  }
284  continue;
285  }
286 
287  if (faceFormat == vertex_normal)
288  {
289  triangle.vertex[0] = GSUtil::fast_a_to_uint64_t(ptr, &ptr) - 1;
290  ptr += 2;
291  if (ptr >= endPtr) continue;
292  triangle.normal[0] = GSUtil::fast_a_to_uint64_t(ptr, &ptr) - 1;
293  if (ptr >= endPtr) continue;
294  triangle.vertex[1] = GSUtil::fast_a_to_uint64_t(ptr, &ptr) - 1;

296  if (ptr >= endPtr) continue;
297  triangle.normal[1] = GSUtil::fast_a_to_uint64_t(ptr, &ptr) - 1;

302  triangle.normal[2] = GSUtil::fast_a_to_uint64_t(ptr, &ptr) - 1;
303  triangle.uv[0] = SIZE_MAX;
304  triangle.uv[1] = SIZE_MAX;

307  ptr++;
308  if (m_badMesh) // currently duplicate the triangle but with reversed winding but this could be improved

312  std::swap(triangle.normal[1], triangle.normal[2]);

314  triangleList.push_back(triangle);
315  }
316  continue;
317  }

320  {
321  triangle.vertex[0] = GSUtil::fast_a_to_uint64_t(ptr, &ptr) - 1;
322  ptr++;
323  if (ptr >= endPtr) continue;
324  triangle.uv[0] = GSUtil::fast_a_to_uint64_t(ptr, &ptr) - 1;
325  if (ptr >= endPtr) continue;
326  triangle.normal[0] = SIZE_MAX;
327  triangle.vertex[1] = GSUtil::fast_a_to_uint64_t(ptr, &ptr) - 1;
328  ptr++;
329  if (ptr >= endPtr) continue;
330  triangle.uv[1] = GSUtil::fast_a_to_uint64_t(ptr, &ptr) - 1;
331  if (ptr >= endPtr) continue;
332  triangle.normal[1] = SIZE_MAX;
333  triangle.vertex[2] = GSUtil::fast_a_to_uint64_t(ptr, &ptr) - 1;
334  ptr++;
335  if (ptr >= endPtr) continue;
336  triangle.uv[2] = GSUtil::fast_a_to_uint64_t(ptr, &ptr) - 1;
337  triangle.normal[2] = SIZE_MAX;
338  triangle.material = currentMaterial;
339  triangleList.push_back(triangle);
340  ptr++;
341  if (m_badMesh) // currently duplicate the triangle but with reversed winding but this could be improved
342  {
343 
344  std::swap(triangle.vertex[1], triangle.vertex[2]);
345  std::swap(triangle.normal[1], triangle.normal[2]);
346  std::swap(triangle.uv[1], triangle.uv[2]);
347  triangleList.push_back(triangle);
348  }
349  continue;
350  }
351 
352  if (faceFormat == vertex_texture_normal)
353  {
354  triangle.vertex[0] = GSUtil::fast_a_to_uint64_t(ptr, &ptr) - 1;
355  ptr++;
356  if (ptr >= endPtr) continue;
357  triangle.uv[0] = GSUtil::fast_a_to_uint64_t(ptr, &ptr) - 1;

364  if (ptr >= endPtr) continue;
365  triangle.uv[1] = GSUtil::fast_a_to_uint64_t(ptr, &ptr) - 1;

367  if (ptr >= endPtr) continue;

393  ptr++;
394  }

478 int FacetedObject::ParseOBJMaterialFile(const std::string &filename, std::map<std::string, OBJMaterial> *materialMap)
479 {
480  DataFile materialsFile;
481  if (materialsFile.ReadFile(filename))
482  {
483  std::string strippedFilename = pystring::strip(filename);
484  if (materialsFile.ReadFile(strippedFilename))
485  {
486  if (strippedFilename[0] == '"' && strippedFilename.back() == '"')
487  {
488  if (materialsFile.ReadFile(strippedFilename.substr(1, strippedFilename.size() - 2)))
489  return __LINE__;
490  }
491  else
492  {
493  return __LINE__;
494  }
495  }
496  }
497 
498  std::string materialsData(materialsFile.GetRawData(), materialsFile.GetSize());
499  std::vector<std::string> lines;
500  pystring::splitlines(materialsData, lines);
501  std::string currentMaterialName;
502  OBJMaterial currentMaterial = {};
503  for (size_t i =0; i < lines.size(); i++)
504  {
505  std::string line = pystring::strip(lines[i]);
506  if (pystring::startswith(line, "#"s)) continue;
507  if (pystring::startswith(line, "newmtl "s))
508  {
509  if (currentMaterialName.size())
510  {
511  (*materialMap)[currentMaterialName] = currentMaterial;
512  currentMaterial = {};
513  }
514  currentMaterialName = pystring::strip(line.substr("newmtl "s.size(), std::string::npos));
515  continue;
516  }
517 
518  std::vector<std::string> tokens;
519  pystring::split(line, tokens);
520  if (tokens.size() >= 2 && tokens[0] == "Ns"s)
521  {
522  currentMaterial.Ns = GSUtil::Double(tokens[1]);
523  continue;
524  }
525  if (tokens.size() >= 4 && tokens[0] == "Ka"s)
526  {
527  currentMaterial.Ka[0] =GSUtil::Double(tokens[1]);
528  currentMaterial.Ka[1] =GSUtil::Double(tokens[2]);
529  currentMaterial.Ka[2] =GSUtil::Double(tokens[3]);
530  continue;
531  }
532  if (tokens.size() >= 4 && tokens[0] == "Kd"s)
533  {
534  currentMaterial.Kd[0] =GSUtil::Double(tokens[1]);
535  currentMaterial.Kd[1] =GSUtil::Double(tokens[2]);
536  currentMaterial.Kd[2] =GSUtil::Double(tokens[3]);
537  continue;
538  }
539  if (tokens.size() >= 4 && tokens[0] == "Ks"s)
540  {
541  currentMaterial.Ks[0] =GSUtil::Double(tokens[1]);
542  currentMaterial.Ks[1] =GSUtil::Double(tokens[2]);
543  currentMaterial.Ks[2] =GSUtil::Double(tokens[3]);
544  continue;
545  }
546  if (tokens.size() >= 4 && tokens[0] == "Ke"s)
547  {
548  currentMaterial.Ke[0] =GSUtil::Double(tokens[1]);
549  currentMaterial.Ke[1] =GSUtil::Double(tokens[2]);
550  currentMaterial.Ke[2] =GSUtil::Double(tokens[3]);
551  continue;
552  }
553  if (tokens.size() >= 2 && tokens[0] == "Ni"s)
554  {
555  currentMaterial.Ni = GSUtil::Double(tokens[1]);
556  continue;
557  }
558  if (tokens.size() >= 2 && tokens[0] == "d"s)
559  {
560  currentMaterial.d = GSUtil::Double(tokens[1]);
561  continue;
562  }
563  if (tokens.size() >= 2 && tokens[0] == "illum"s)
564  {
565  currentMaterial.illum = GSUtil::Int(tokens[1]);
566  continue;
567  }
568  }
569  if (currentMaterialName.size())
570  {
571  (*materialMap)[currentMaterialName] = currentMaterial;
572  currentMaterial = {};
573  }
574  return 0;
575 }

577 int FacetedObject::ParsePLYFile(const std::string &filename)
578 {
579  m_filename = filename;

1291 // utility to calculate a face normal
1292 // this assumes anticlockwise winding
1293 // for a triangle p1, p2, p3, if the vector A = p2 - p1 and the vector B = p3 - p1
1294 // then the normal N = A x B and can be calculated by:
1295 // this can be simplified to:
1296 // Nx = Ay * Bz - Az * By
1297 // Ny = Az * Bx - Ax * Bz
1298 // Nz = Ax * By - Ay * Bx
1299 void FacetedObject::ComputeFaceNormal(const double *v1, const double *v2, const double *v3, double normal[3])
1300 {
1301  double a[3], b[3];
1302 
1303  // calculate in plane vectors
1304  a[0] = v2[0] - v1[0];
1305  a[1] = v2[1] - v1[1];
1306  a[2] = v2[2] - v1[2];
1307  b[0] = v3[0] - v1[0];
1308  b[1] = v3[1] - v1[1];
1309  b[2] = v3[2] - v1[2];
1310 
1311  // cross(a, b, normal);
1312  normal[0] = a[1] * b[2] - a[2] * b[1];
1313  normal[1] = a[2] * b[0] - a[0] * b[2];
1314  normal[2] = a[0] * b[1] - a[1] * b[0];
1315 
1316  // normalize(normal);
1317  double norm = sqrt(normal[0] * normal[0] +
1318  normal[1] * normal[1] +
1319  normal[2] * normal[2]);
1320 
1321  if (norm > 0.0)
1322  {
1323  normal[0] /= norm;
1324  normal[1] /= norm;
1325  normal[2] /= norm;
1326  }
