20 FacetedPolyline::FacetedPolyline(std::vector<pgd::Vector3> *polyline, double radius, size_t n, const QColor &blendColour, double blendFraction, bool internal)
21 {
22  setBlendColour(blendColour, blendFraction);
23  if (internal)
24  {
25  std::vector<pgd::Vector3> profile;
26  AllocateMemory(n * (polyline->size() * 2 + 2));
27 
28  // need to add extra tails to the polyline for direction padding
29  std::vector<pgd::Vector3> newPolyline;
30  newPolyline.reserve(polyline->size() + 2);
31  pgd::Vector3 v0 = (*polyline)[1] - (*polyline)[0];
32  pgd::Vector3 v1 = (*polyline)[0] - v0;
33  newPolyline.push_back(v1);
34  for (size_t i = 0; i < polyline->size(); i++) newPolyline.push_back((*polyline)[i]);
35  v0 = (*polyline)[polyline->size() - 1] - (*polyline)[polyline->size() - 2];
36  v1 = (*polyline)[polyline->size() - 1] + v0;
37  newPolyline.push_back(v1);
38 
39  // create the profile
40  double delTheta = 2 * M_PI / n;
41  double theta = M_PI / 2;
42  for (size_t i = 0; i < n; i++)
43  {
44  v0.x = cos(theta) * radius;
45  v0.y = sin(theta) * radius;
46  v0.z = 0;
47  theta -= delTheta;
48  profile.push_back(v0);
49  }
50 
51  Extrude(&newPolyline, &profile);
52  }
53  else
54  {
55  gleSetNumSides(int(n));
56  size_t nPoints = polyline->size() + 2;
57  auto point_array = std::make_unique<gleDouble[][3]>(nPoints);

62  v0 = (*polyline)[polyline->size() - 1] - (*polyline)[polyline->size() - 2];
63  v1 = (*polyline)[polyline->size() - 1] + v0;
64  point_array[nPoints][0] = v1.x; point_array[nPoints][1] = v1.y; point_array[nPoints][2] = v1.z;
65  auto color_array = std::make_unique<gleColor[]>(nPoints);
66  float r = blendColour.redF();
67  float g = blendColour.greenF();
68  float b = blendColour.blueF();

89 // extrude profile along a poly line using sharp corners
90 // profile is a 2D shape with z = 0 for all values.
91 // polyline needs to have no parallel neighbouring segements
92 // anti-clockwise winding assumed (I think)
93 // first and last point of polyline used for direction only!
94 void FacetedPolyline::Extrude(std::vector<pgd::Vector3> *polyline, std::vector<pgd::Vector3> *profile)
95 {
96  unsigned int i, j;
97  Line3D line;
98  pgd::Vector3 v1, v2, p1, p2;
99  const double epsilon = 0.000001;
100  const double epsilon2 = epsilon * epsilon;
101  auto polygon = std::make_unique<double []>(profile->size() * 3);
102  auto polygonNormal = std::make_unique<double []>(profile->size() * 3);
103 
104  // define the planes of the joins
105  std::vector<Plane3D> joinPlanes;
106  joinPlanes.reserve(polyline->size());
107  Plane3D plane;
108  for (i = 1; i < (*polyline).size() - 1; i++)
109  {
110  v1 = (*polyline)[i] - (*polyline)[i - 1];
111  v2 = (*polyline)[i + 1] - (*polyline)[i];
112  v1.Normalize();
113  v2.Normalize();
114  p1 = v1 - v2;
115  if (p1.Magnitude2() > epsilon2) // not parallel so use two vector form of plane
116  {
117  p2 = v1 ^ v2;
118  plane = Plane3D(&(*polyline)[i], &p1, &p2);
119  }
120  else // parallel so use normal-point form
121  {
122  plane = Plane3D(&(*polyline)[i], &v1);
123  }
124 
125  joinPlanes.push_back(plane);
126  }
127 
128  // define the rotated (*profile) for the first line segment
129  // note for a truly generic routine you need two rotations to allow an up vector to be defined
130  pgd::Vector3 zVec(0, 0, 1);
131  v1 = (*polyline)[1] - (*polyline)[0];
132  pgd::Quaternion q = pgd::FindRotation(zVec, v1);
133  std::vector<pgd::Vector3> rotatedProfile;
134  rotatedProfile.reserve(polyline->size());
135  for (i = 0; i < (*profile).size(); i++)
136  {
137  pgd::Vector3 v = pgd::QVRotate(q, (*profile)[i]);
138  rotatedProfile.push_back(v);
139  }
140 
141  // find the intersections on the join planes
142  std::vector<pgd::Vector3> vertexList;
143  vertexList.reserve(rotatedProfile.size() * joinPlanes.size());
144  for (i = 0; i < rotatedProfile.size(); i++)
145  {
146  v2 = (*polyline)[0] + rotatedProfile[i];
147  line = Line3D(&v2, &v1);
148  for (j = 0; j < joinPlanes.size(); j++)
149  {
150  if (Intersection(&line, &joinPlanes[j], &v2))
151  {
152  vertexList.push_back(v2);
153  if (j < joinPlanes.size() - 1)
154  {
155  p1 = (*polyline)[j + 2] - (*polyline)[j + 1];
156  line = Line3D(&v2, &p1);
157  }
158  }
159  else
160  {
161  qDebug() << "Error finding line & plane intersection" << "\n";
162  return;
163  }
164  }
165  }
166 
167  // now construct faces
168  pgd::Vector3 centroid1, centroid2, normal;
169  for (i = 0; i < rotatedProfile.size(); i++)
170  {
171  for (j = 0; j < joinPlanes.size() - 1; j++)
172  {
173  centroid1 = polyline->at(j + 1);
174  centroid2 = polyline->at(j + 2);
175  if (i < rotatedProfile.size() - 1)
176  {
177  polygon[0] = vertexList[j + joinPlanes.size() * i].x;
178  polygon[1] = vertexList[j + joinPlanes.size() * i].y;
179  polygon[2] = vertexList[j + joinPlanes.size() * i].z;
180  polygon[3] = vertexList[1 + j + joinPlanes.size() * i].x;
181  polygon[4] = vertexList[1 + j + joinPlanes.size() * i].y;
182  polygon[5] = vertexList[1 + j + joinPlanes.size() * i].z;
183  polygon[6] = vertexList[1 + j + joinPlanes.size() * (i + 1)].x;
184  polygon[7] = vertexList[1 + j + joinPlanes.size() * (i + 1)].y;
185  polygon[8] = vertexList[1 + j + joinPlanes.size() * (i + 1)].z;
186  polygon[9] = vertexList[j + joinPlanes.size() * (i + 1)].x;
187  polygon[10] = vertexList[j + joinPlanes.size() * (i + 1)].y;
188  polygon[11] = vertexList[j + joinPlanes.size() * (i + 1)].z;
189  }
190  else
191  {
192  polygon[0] = vertexList[j + joinPlanes.size() * i].x;
193  polygon[1] = vertexList[j + joinPlanes.size() * i].y;
194  polygon[2] = vertexList[j + joinPlanes.size() * i].z;
195  polygon[3] = vertexList[1 + j + joinPlanes.size() * i].x;
196  polygon[4] = vertexList[1 + j + joinPlanes.size() * i].y;
197  polygon[5] = vertexList[1 + j + joinPlanes.size() * i].z;
198  polygon[6] = vertexList[1 + j].x;
199  polygon[7] = vertexList[1 + j].y;
200  polygon[8] = vertexList[1 + j].z;
201  polygon[9] = vertexList[j].x;
202  polygon[10] = vertexList[j].y;
203  polygon[11] = vertexList[j].z;
204  }
205  normal = pgd::Vector3(polygon[0], polygon[1], polygon[2]) - centroid1;
206  normal.Normalize();
207  polygonNormal[0] = normal.x;
208  polygonNormal[1] = normal.y;
209  polygonNormal[2] = normal.z;
210  normal = pgd::Vector3(polygon[3], polygon[4], polygon[5]) - centroid2;
211  normal.Normalize();
212  polygonNormal[3] = normal.x;
213  polygonNormal[4] = normal.y;
214  polygonNormal[5] = normal.z;
215  normal = pgd::Vector3(polygon[6], polygon[7], polygon[8]) - centroid2;
216  normal.Normalize();
217  polygonNormal[6] = normal.x;
218  polygonNormal[7] = normal.y;
219  polygonNormal[8] = normal.z;
220  normal = pgd::Vector3(polygon[9], polygon[10], polygon[11]) - centroid1;
221  normal.Normalize();
222  polygonNormal[9] = normal.x;
223  polygonNormal[10] = normal.y;
224  polygonNormal[11] = normal.z;
225  this->AddPolygon(polygon.get(), 4, polygonNormal.get());
226  }
227  }
228 
229  // end caps
230  for (i = 0; i < rotatedProfile.size(); i++)
231  {
232  polygon[i * 3] = vertexList[joinPlanes.size() * i].x;
233  polygon[i * 3 + 1] = vertexList[joinPlanes.size() * i].y;
234  polygon[i * 3 + 2] = vertexList[joinPlanes.size() * i].z;
235  }
236  this->AddPolygon(polygon.get(), rotatedProfile.size());
237  for (i = 0; i < rotatedProfile.size(); i++)
238  {
239  polygon[i * 3] = vertexList[joinPlanes.size() - 1 + joinPlanes.size() * (rotatedProfile.size() - i - 1)].x;
240  polygon[i * 3 + 1] = vertexList[joinPlanes.size() - 1 + joinPlanes.size() * (rotatedProfile.size() - i - 1)].y;
241  polygon[i * 3 + 2] = vertexList[joinPlanes.size() - 1 + joinPlanes.size() * (rotatedProfile.size() - i - 1)].z;
242  }
243  this->AddPolygon(polygon.get(), rotatedProfile.size());
244 }

246 // find intersection of line and plane
247 // returns true on success, false if no intersection
248 bool FacetedPolyline::Intersection(Line3D *line, Plane3D *plane, pgd::Vector3 *intersection)
249 {
250  double denominator = line->direction * plane->GetNormal();
251  const double epsilon = 0.000001;
252  double t;
253 
254  if (fabs(denominator) < epsilon)
255  {
256  // line and plane very close to parallel so they probably don't meet
257  // but perhaps the origin is in the plane
258  if (fabs(line->origin.x * plane->a + line->origin.y * plane->b + line->origin.z * plane->c + plane->d) > epsilon)
259  {
260  t = 0;
261  *intersection = line->origin;
262  return true;
263  }
264  else
265  {
266  return false;
267  }
268  }
269 
270  // compute intersection
271 
272  t = -(plane->a * line->origin.x + plane->b * line->origin.y + plane->c * line->origin.z + plane->d);
273  t = t / denominator;
274  *intersection = line->origin + t * line->direction;
275 
276  return true;
277 
278 }
