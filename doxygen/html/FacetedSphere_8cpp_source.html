10 // this code largely based on code by Jon Leech from:
11 // http://www.gamedev.net/reference/articles/article427.asp
12 /*
13  * sphere - generate a polygon mesh approximating a sphere by
14  * recursive subdivision. First approximation is an octahedron;
15  * each level of refinement increases the number of polygons by
16  * a factor of 4.
17  * Level 3 (128 polygons) is a good tradeoff if gouraud
18  * shading is used to render the database.
19  *
20  * Usage: sphere [level]
21  * level is an integer >= 1 setting the recursion level (default 1).
22  *
23  * Notes:
24  *
25  * The triangles are generated with vertices in clockwise order as
26  * viewed from the outside in a right-handed coordinate system.
27  * To reverse the order, compile with COUNTERCLOCKWISE defined.
28  *
29  * Shared vertices are not retained, so numerical errors may produce
30  * cracks between polygons at high subdivision levels.
31  *
32  * The subroutines print_object() and print_triangle() should
33  * be changed to generate whatever the desired database format is.
34  * If UNC is defined, a PPHIGS text archive is generated.
35  *
36  * Jon Leech 3/24/89
37  */

77 FacetedSphere::FacetedSphere(double radius, size_t maxlevels, const QColor &blendColour, double blendFraction)
78 {
79  setBlendColour(blendColour, blendFraction);
80 
81  /* Vertices of a unit octahedron */
82  triangle octahedron[] =
83  {
84  {{ XPLUS, ZPLUS, YPLUS }, 0.0},
85  {{ YPLUS, ZPLUS, XMIN }, 0.0},
86  {{ XMIN, ZPLUS, YMIN }, 0.0},
87  {{ YMIN, ZPLUS, XPLUS }, 0.0},
88  {{ XPLUS, YPLUS, ZMIN }, 0.0},
89  {{ YPLUS, XMIN, ZMIN }, 0.0},
90  {{ XMIN, YMIN, ZMIN }, 0.0},
91  {{ YMIN, XPLUS, ZMIN }, 0.0}
92  };
93 
94  /* An octahedron */
95  object oct =
96  {
97  sizeof(octahedron) / sizeof(octahedron[0]),
98  &octahedron[0]
99  };
100 
101  object *oldObject, *newObject;
102  size_t i, level;
103 
104  if (maxlevels < 1) maxlevels = 1;
105  m_Level = maxlevels;
106  m_Radius = radius;
107 
108 #ifdef COUNTERCLOCKWISE
109  /* Reverse order of points in each triangle */
110  for (i = 0; i < oct.npoly; i++)
111  {
112  point tmp;
113  tmp = oct.poly[i].pt[0];
114  oct.poly[i].pt[0] = oct.poly[i].pt[2];
115  oct.poly[i].pt[2] = tmp;
116  }
117 #endif
118 
119  oldObject = &oct;
120 
121  /* Subdivide each starting triangle (maxlevels - 1) times */
122  for (level = 1; level < maxlevels; level++)
123  {
124  /* Allocate a new object */
125  newObject = static_cast<object *>(malloc(sizeof(object)));
126  if (newObject == nullptr)
127  {
128  fprintf(stderr, "FacetedSphere: Out of memory on subdivision level %zud\n", level);
129  exit(1);
130  }
131  newObject->npoly = oldObject->npoly * 4;
132 
133  /* Allocate 4* the number of points in the current approximation */
134  newObject->poly = static_cast<triangle *>(malloc(newObject->npoly * sizeof(triangle)));
135  if (newObject->poly == nullptr)
136  {
137  fprintf(stderr, "FacetedSphere: Out of memory on subdivision level %zu\n", level);
138  exit(1);
139  }
140 
141  /* Subdivide each polygon in the old approximation and normalize
142  * the new points thus generated to lie on the surface of the unit
143  * sphere.
144  * Each input triangle with vertices labelled [0,1,2] as shown
145  * below will be turned into four new triangles:
146  *
147  * Make new points
148  * a = (0+2)/2
149  * b = (0+1)/2
150  * c = (1+2)/2
151  * 1
152  * /\ Normalize a, b, c
153  * / \
154  * b/____\ c Construct new triangles
155  * /\ /\ [0,b,a]
156  * / \ / \ [b,1,c]
157  * /____\/____\ [a,b,c]
158  * 0 a 2 [a,c,2]
159  */
160  for (i = 0; i < oldObject->npoly; i++)
161  {
162  triangle *oldt = &oldObject->poly[i];
163  triangle *newt = &newObject->poly[i * 4];
164  point a, b, c;
165 
166  a = *normalize(midpoint(&oldt->pt[0], &oldt->pt[2]));
167  b = *normalize(midpoint(&oldt->pt[0], &oldt->pt[1]));
168  c = *normalize(midpoint(&oldt->pt[1], &oldt->pt[2]));
169 
170  newt->pt[0] = oldt->pt[0];
171  newt->pt[1] = b;
172  newt->pt[2] = a;
173  newt++;
174 
175  newt->pt[0] = b;
176  newt->pt[1] = oldt->pt[1];
177  newt->pt[2] = c;
178  newt++;
179 
180  newt->pt[0] = a;
181  newt->pt[1] = b;
182  newt->pt[2] = c;
183  newt++;
184 
185  newt->pt[0] = a;
186  newt->pt[1] = c;
187  newt->pt[2] = oldt->pt[2];
188  }
189 
190  if (level > 1)
191  {
192  free(oldObject->poly);
193  free(oldObject);
194  }
195 
196  /* Continue subdividing new triangles */
197  oldObject = newObject;
198  }
199 
200  /* Print out resulting approximation */
201  //print_object(oldObject, maxlevels);
202 
203  // add the faces to the Faceted Object
204  double vertex[9];
205  AllocateMemory(oldObject->npoly);
206  for (i = 0; i < oldObject->npoly; i++)
207  {
208  vertex[0] = oldObject->poly[i].pt[0].x;
209  vertex[1] = oldObject->poly[i].pt[0].y;
210  vertex[2] = oldObject->poly[i].pt[0].z;
211  vertex[3] = oldObject->poly[i].pt[1].x;
212  vertex[4] = oldObject->poly[i].pt[1].y;
213  vertex[5] = oldObject->poly[i].pt[1].z;
214  vertex[6] = oldObject->poly[i].pt[2].x;
215  vertex[7] = oldObject->poly[i].pt[2].y;
216  vertex[8] = oldObject->poly[i].pt[2].z;
217  AddTriangle(vertex);
218  }
219 
220  free(oldObject->poly);
221  free(oldObject);
222 
223  Scale(radius, radius, radius);
224 
225 // qDebug() << "FacetedSphere " << GetNumTriangles() << " triangles created\n";
226 }

244 size_t FacetedSphere::EstimateLevel(size_t requestedFaces, size_t *actualFaces)
245 {
246  // basic octohedron has 8 faces (level 1)
247  // and this is doubled each level
248  // n = 8 x 2^(level - 1)
249  double actualLevel = 1 + (std::log(double(requestedFaces)) - std::log(8.0)) / std::log(2);
250  size_t level = size_t(std::ceil(actualLevel));
251  if (actualFaces) *actualFaces = 8 * size_t(std::pow(2.0, double(level)) + 0.5);
252  return level;
253 }

255 /* Normalize a point p */
256 point *normalize(point *p)
257 {
258  static point r;
259  double mag;
260 
261  r = *p;
262  mag = r.x * r.x + r.y * r.y + r.z * r.z;
263  if (mag != 0.0)
264  {
265  mag = 1.0 / sqrt(mag);
266  r.x *= mag;
267  r.y *= mag;
268  r.z *= mag;
269  }
270 
271  return &r;
272 }

274 /* Return the average of two points */
275 point *midpoint(point *a, point *b)
276 {
277  static point r;
278 
279  r.x = (a->x + b->x) * 0.5;
280  r.y = (a->y + b->y) * 0.5;
281  r.z = (a->z + b->z) * 0.5;
282 
283  return &r;
284 }
