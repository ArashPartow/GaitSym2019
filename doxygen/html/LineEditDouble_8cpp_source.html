101 void LineEditDouble::setValue(double value)
102 {
103  // do some checks for validity
104  while (!std::isfinite(value))
105  {
106  if (value == std::numeric_limits<double>::infinity()) { value = std::numeric_limits<double>::max(); break; }
107  if (value == -std::numeric_limits<double>::infinity()) { value = -std::numeric_limits<double>::max(); break; }
108  value = 0.0;
109  break;
110  }
111  // there are potential round trip problems here if value is equal to either bottom or top
112  const DoubleValidator *validator = static_cast<const DoubleValidator *>(this->validator());
113  QString valueString;
114  int decimals = validator->decimals();
115  valueString.setNum(value, 'g', decimals);
116  int pos = 0;
117  if (validator->validate(valueString, pos) != QValidator::Acceptable)
118  {
119  if (decimals > std::numeric_limits<double>::max_digits10) valueString.setNum(value, 'e', std::numeric_limits<double>::max_digits10);
120  else valueString.setNum(value, 'e', decimals);
121  }
122  this->setText(valueString);
123 }
