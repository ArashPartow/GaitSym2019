28 void MarkerEllipseDriver::Initialise(double omega, double sigma, double XR, double YR, double phi, Marker *markerEllipseCentre, Marker *markerEllipseRim, DataTarget *phaseControlInput)
29 {
30  m_omega = omega; // rad s-1 intrinsic angular velocity
31  m_sigma = sigma; // gain for the phase correction
32  m_XR = XR; // m x-direction radius
33  m_YR = YR; // m y-direction radius
34  m_phi = phi; // rad initial phase on LF leg's oscillator
35  m_markerEllipseCentre = markerEllipseCentre; // this marker defines the centre position and local coordinate system for the controller
36 
37  // and set the derived values
38  m_phiDot = m_omega;
39  m_X = m_XR * std::cos(m_phi);
40  m_Y = m_YR * std::sin(m_phi);
41 
42  pgd::Quaternion rimLocalQ = m_markerEllipseCentre->GetQuaternion();
43  pgd::Vector3 rimWorldP = m_markerEllipseCentre->GetWorldPosition(pgd::Vector3(m_X, m_Y, 0));
44  m_markerEllipseRim = markerEllipseRim;
45  m_markerEllipseRim->SetQuaternion(rimLocalQ.n, rimLocalQ.x, rimLocalQ.y, rimLocalQ.z);
46  m_markerEllipseRim->SetWorldPosition(rimWorldP.x ,rimWorldP.y, rimWorldP.z);
47  m_phaseControlInput = phaseControlInput;
48 }

58 void MarkerEllipseDriver::Update()
59 {
60  assert(simulation()->GetStepCount() == lastStepCount() + 1);
61  setLastStepCount(simulation()->GetStepCount());
62 
63  if (m_omegaDriver) m_omega = m_omegaDriver->value();
64  if (m_sigmaDriver) m_sigma = m_sigmaDriver->value();
65  if (m_XRDriver) m_XR = m_XRDriver->value();
66  if (m_YRDriver) m_YR = m_YRDriver->value();
67 
68  // main control algorithm
69  if (!m_phaseControlInput)
70  {
71  m_phiDot = m_omega;
72  }
73  else
74  {
75  // we need to do something to correct for the phase
76  m_valueChangeDirection = detectSignChange(m_phaseControlInput->calculateError(simulation()->GetTime()));
77  if (m_valueChangeDirection != 0)
78  {
79  m_halfPeriod = simulation()->GetTime() - m_lastPhaseChangeTime;
80  m_lastPhaseChangeTime = simulation()->GetTime();
81  m_phiDot = std::clamp(M_PI / (m_halfPeriod * m_periodMultiplier), 0.0, m_maxPhiDot); // this copes with halfPeriod of zero since divide by zero is +/- infinity
82  // but we need to tweak m_phiDot to get the phase relationship eventually
83  if (m_valueChangeDirection > 0) m_wantedPhi = std::fmod(2 * M_PI + std::fmod(M_PI_2 + m_phaseOffset, 2 * M_PI), 2 * M_PI);
84  else m_wantedPhi = std::fmod(2 * M_PI + std::fmod(3 * M_PI_2 + m_phaseOffset, 2 * M_PI), 2 * M_PI);
85  m_delPhi = m_wantedPhi - m_phi;
86  if (std::fabs(m_delPhi) > M_PI) m_delPhi = 2 * M_PI + m_phi - m_wantedPhi;
87  m_phiDot += m_sigma * -m_delPhi; // this is a P controller. A PID controller might be better (certainly a PD controller)
88  }
89  }
90 
91  // update m_phi depending on m_phi_dot values
92  m_phi = std::fmod(2 * M_PI + std::fmod(m_phi + m_phiDot * simulation()->GetTimeIncrement(), 2 * M_PI), 2 * M_PI); // do fmod twice to get a value from 0 to 2pi
93 
94  m_X = m_XR * std::cos(m_phi);
95  m_Y = m_YR * std::sin(m_phi);
96 
97  // get the world position of the MarkerEllipse target
98  pgd::Quaternion rimLocalQ = m_markerEllipseCentre->GetQuaternion();
99  pgd::Vector3 rimWorldP = m_markerEllipseCentre->GetWorldPosition(pgd::Vector3(m_X, m_Y, 0));
100  m_markerEllipseRim->SetQuaternion(rimLocalQ.n, rimLocalQ.x, rimLocalQ.y, rimLocalQ.z);
101  m_markerEllipseRim->SetWorldPosition(rimWorldP.x ,rimWorldP.y, rimWorldP.z);
102 
103 }

105 int MarkerEllipseDriver::detectSignChange(double value)
106 {
107  double lastValue = m_butterworthFilter.Output();
108  m_butterworthFilter.AddNewSample(value);
109  double delta = m_butterworthFilter.Output() - lastValue;
110 // std::cerr << "delta = " << delta << "\n";
111  if (m_phaseStateIncreasing && delta > 0)
112  {
113  m_phaseStateChangeCount = 0;
114  return 0;
115  }
116  if (!m_phaseStateIncreasing && delta < 0)
117  {
118  m_phaseStateChangeCount = 0;
119  return 0;
120  }
121  if (m_phaseStateIncreasing && delta < 0)
122  {
123  m_phaseStateChangeCount++;
124  if (m_phaseStateChangeCount > m_phaseStateCountThreshold)
125  {
126  m_phaseStateIncreasing = false;
127  m_phaseStateChangeCount = 0;
128  return -1;
129  }
130  return 0;
131  }
132  if (!m_phaseStateIncreasing && delta > 0)
133  {
134  m_phaseStateChangeCount++;
135  if (m_phaseStateChangeCount > m_phaseStateCountThreshold)
136  {
137  m_phaseStateIncreasing = true;
138  m_phaseStateChangeCount = 0;
139  return +1;
140  }
141  return 0;
142  }
143  return 0;
144 }
