35 void PIDErrorInController::Update()
36 {
37  assert(simulation()->GetStepCount() == lastStepCount() + 1);
38  setLastStepCount(simulation()->GetStepCount());
39 
40  m_dt = simulation()->GetTimeIncrement();
41 
42  // in this driver, the error is driven by the upstream driver
43  m_error = dataSum();
44  if (m_previous_error == DBL_MAX) m_previous_error = m_error;
45 
46  // do the PID calculations
47  m_integral = m_integral + (m_error * m_dt);
48  m_derivative = (m_error - m_previous_error) / m_dt;
49  m_output = (m_Kp * m_error) + (m_Ki * m_integral) + (m_Kd * m_derivative);
50  m_previous_error = m_error;
51 
52  // now set the output based on the PID output
53  // note that we limit the value to the range
54  setValue(Clamp(m_output));
55 }
