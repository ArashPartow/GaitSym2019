31 PlaneGeom::PlaneGeom(dSpaceID space, double a, double b, double c, double d)
32 {
33  // create the geom
34  double length = std::sqrt(a * a + b * b + c * c);
35  if (length < std::numeric_limits<double>::min()) // standard fixup
36  {
37  a = 0;
38  b = 0;
39  c = 1;
40  d = 0;
41  }
42  else
43  {
44  a = a / length;
45  b = b / length;
46  c = c / length;
47  }
48  setGeomID(dCreatePlane(space, a, b, c, d));
49  dGeomSetData(GetGeomID(), this);
50 
51 #ifdef USE_QT_IRRLICHT
52  m_floorSceneNode = 0;
53 
54  //glGenTextures(1, &m_textureID);
55  m_texture = 0;
56  m_trackDepth = 0;
57  m_lastDisplayTime = -1;
58  m_nx = 0;
59  m_ny = 0;
60  Colour mappedColour;
61  for (int i = 0; i < 256; i++)
62  {
63  GLUtils::SetColourFromMap(float(i) / 255.0, JetColourMap, &mappedColour);
64  m_colourMap[i][0] = int(0.5 + mappedColour.r * 255.0); // red
65  m_colourMap[i][1] = int(0.5 + mappedColour.g * 255.0); // green
66  m_colourMap[i][2] = int(0.5 + mappedColour.b * 255.0); // blue
67  m_colourMap[i][3] = int(0.5 + mappedColour.alpha * 255.0); // alpha
68  }
69 
70  m_planeOrigin = pgd::Vector3(a * d, b * d, c * d); // this is in world coordinates
71  // now find a rotation that aligns the world Z axis with the plane normal
72  m_planeOrientation = FindRotation(pgd::Vector3(0, 0, 1), pgd::Vector3(a, b, c));
73  m_checkerboardLow = 0;
74  m_checkerboardHigh = 0;
75  m_trackDrawThreshold = 0;
76 
77  m_floorTexture = ":/images/Floor Textures/floor_checker_2.png";
78 #endif
79

91 std::string *PlaneGeom::createFromAttributes()
92 {
93  if (Geom::createFromAttributes()) return lastErrorPtr();
94  std::string buf;
95 
96  if (geomMarker()->GetBody())
97  {
98  setLastError("GEOM ID=\""s + name() +"\" PlaneGeom must be attached to the World"s);
99  return lastErrorPtr();
100  }
101 
102  // because planes are non-placeable we need to calculate the plane equation directly from the marker
103  //
104  // we want the plane in its cartesian form a*x+b*y+c*z = d
105  // the marker gives a point on the plane, 2 vectors in the plane (x & y axes)
106  // and the plane normal (z axes)
107  // the normal is (a,b,c)
108  // d is the dot product of the normal at the point on the plane
109 
110  pgd::Vector3 normal = geomMarker()->GetWorldAxis(Marker::Axis::Z);
111  pgd::Vector3 point = geomMarker()->GetWorldPosition();
112  double a = normal.x;
113  double b = normal.y;
114  double c = normal.z;
115  double d = normal.Dot(point);
116  dGeomPlaneSetParams(GetGeomID(), a, b, c, d);
117  return nullptr;
118 }

139 void PlaneGeom::SetTrackPatch(double trackPatchStartX, double trackPatchStartY, double trackPatchEndX, double trackPatchEndY, double trackPatchResolutionX, double trackPatchResolutionY)
140 {
141  m_trackPatchStartX = trackPatchStartX;
142  m_trackPatchEndX = trackPatchEndX;
143  m_trackPatchResolutionX = trackPatchResolutionX;
144  m_trackPatchStartY = trackPatchStartY;
145  m_trackPatchEndY = trackPatchEndY;
146  m_trackPatchResolutionY = trackPatchResolutionY;
147 
148  m_nx = (int)(0.00001 + (m_trackPatchEndX - m_trackPatchStartX) / m_trackPatchResolutionX);
149  m_ny = (int)(0.00001 + (m_trackPatchEndY - m_trackPatchStartY) / m_trackPatchResolutionY);
150  int textureSize = m_nx * m_ny * 4; // RGBA format
151  if (textureSize)
152  {
153  m_texture = new unsigned char[textureSize];
154  std::fill_n(m_texture, textureSize, 0);
155  m_trackDepth = new double[textureSize];
156  if (m_checkerboardHigh == m_checkerboardLow)
157  {
158  std::fill_n(m_trackDepth, textureSize, m_checkerboardLow);
159  }
160  else
161  {
162  double *trackDepthPtr = m_trackDepth;
163  for (int iy = 0; iy < m_ny; iy++)
164  {
165  for (int ix = 0; ix < m_nx; ix++)
166  {
167  if (iy % 2)
168  {
169  if (ix %2) *trackDepthPtr++ = m_checkerboardHigh;
170  else *trackDepthPtr++ = m_checkerboardLow;
171  }
172  else
173  {
174  if (ix %2) *trackDepthPtr++ = m_checkerboardLow;
175  else *trackDepthPtr++ = m_checkerboardHigh;
176  }
177  }
178  }
179  }
180  }

183 // this routine adds a impulse to the plane impulse store to emulate the appearance of footprints
184 //.the values are in world coordinates
185 void PlaneGeom::AddImpulse(double x, double y, double z, double fx, double fy, double fz, double time)
186 {
187  if (m_trackDepth == 0) return;
188  pgd::Vector3 offsetVec = pgd::Vector3(x, y, z) - m_planeOrigin;
189  pgd::Vector3 posOnPlane = QVRotate(m_planeOrientation, offsetVec);
190  pgd::Vector3 force = QVRotate(m_planeOrientation, pgd::Vector3(fx, fy, fz));
191  pgd::Vector3 impulse = force * time;
192  int ix = (int)((posOnPlane.x - m_trackPatchStartX) / m_trackPatchResolutionX);
193  int iy = (int)((posOnPlane.y - m_trackPatchStartY) / m_trackPatchResolutionY);
194  if (ix >= 0 && x < m_nx && iy >= 0 && y < m_ny)
195  {
196  double *ptr = m_trackDepth + iy * m_nx + ix;
197  *ptr += impulse.z;
198  // std::cerr << "ix = " << ix << " iy = " << iy << " Impulse = " << *ptr << "\n";
199  }
