140 void SimulationWidget::paintGL()
141 {
142  QPainter painter;
143  painter.begin(this);
144 
145  painter.beginNativePainting();
146 
147  QOpenGLVertexArrayObject::Binder vaoBinder(m_vao);
148 
149  glClearColor(GLclampf(m_backgroundColour.redF()), GLclampf(m_backgroundColour.greenF()), GLclampf(m_backgroundColour.blueF()), GLclampf(m_backgroundColour.alphaF()));
150  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
151  glEnable(GL_DEPTH_TEST);
152  // glEnable(GL_CULL_FACE);
153 
154  // set the projection matrix
155  float aspectRatio = float(width()) / float(height());
156  float viewHeight = 2.0f * std::sin((m_FOV / 2.0f) * float(M_PI) / 180.0f) * m_cameraDistance;
157  float viewWidth = viewHeight * aspectRatio;
158 
159  m_proj.setToIdentity();
160  if (m_orthographicProjection)
161  {
162  m_proj.ortho(-viewWidth, viewWidth, -viewHeight, viewHeight, m_frontClip, m_backClip); // multiply by orthographic projection matrix
163  }
164  else
165  {
166  m_proj.perspective(m_FOV, aspectRatio, m_frontClip, m_backClip); // multiply by perspective projection
167  }
168 
169  // set the view matrix
170  m_view.setToIdentity();
171  QVector3D eye(m_COIx - m_cameraVecX * m_cameraDistance, m_COIy - m_cameraVecY * m_cameraDistance, m_COIz - m_cameraVecZ * m_cameraDistance);
172  QVector3D centre(m_COIx, m_COIy, m_COIz);
173  QVector3D up(m_upX, m_upY, m_upZ);
174  m_view.lookAt(eye, centre, up);
175 
176  // now draw things
177  if (m_simulation)
178  {
179  drawModel();
180  }
181 
182 // for (auto &&iter : m_extraObjectsToDrawMap)
183 // {
184 // iter.second->Draw();
185 // }
186 
187  // the 3d cursor
188  // qDebug() << "Cursor " << m_cursor3DPosition.x() << " " << m_cursor3DPosition.y() << " " << m_cursor3DPosition.z();
189  m_cursor3D->SetDisplayPosition(double(m_cursor3DPosition.x()), double(m_cursor3DPosition.y()), double(m_cursor3DPosition.z()));
190  m_cursor3D->SetDisplayScale(double(m_cursorRadius), double(m_cursorRadius), double(m_cursorRadius));
191  m_cursor3D->setSimulationWidget(this);
192  m_cursor3D->Draw();
193 
194  // the global axes
195  m_globalAxes->SetDisplayPosition(0, 0, 0);
196  m_globalAxes->SetDisplayScale(double(m_axesScale), double(m_axesScale), double(m_axesScale));
197  m_globalAxes->setSimulationWidget(this);
198  m_globalAxes->Draw();
199 
200  // any manipulation feedback?
201 
202  // raster mode positioning
203  // with origin at top left
204  glDisable(GL_DEPTH_TEST);
205  StrokeFont strokeFont;
206  const GLfloat threshold = 105.0f / 255.0f;
207  GLfloat backgroundDelta = GLfloat((m_backgroundColour.redF() * 0.299) + (m_backgroundColour.greenF() * 0.587) + (m_backgroundColour.blueF() * 0.114));
208  if (backgroundDelta > threshold) strokeFont.SetRGBA(0, 0, 0, 1);
209  else strokeFont.SetRGBA(1, 1, 1, 1);
210  strokeFont.setGlWidget(this);
211  QMatrix4x4 lineVP;
212  lineVP.ortho(0, float(width()), 0, float(height()), -1, 1);
213  strokeFont.setVpMatrix(lineVP);
214  glLineWidth(2); // this doesn't seem to work on the Mac
215 
216  if (m_trackballFlag && m_trackball->GetOutsideRadius())
217  {
218  float centreX = float(width()) / 2;
219  float centreY = float(height()) / 2;
220  float radius = float(m_trackball->GetTrackballRadius());
221  strokeFont.AddCircle(centreX, centreY, 0, radius, 180);
222  }
223 
224  strokeFont.Draw();
225 
226  painter.endNativePainting();
227  painter.end();
228 }

244 void SimulationWidget::mousePressEvent(QMouseEvent *event)
245 {
246  // on high resolution (e.g.retina) displays the units of the viewport are device pixels whereas the units of event->pos() are scaled pixels
247  // the following mapping should always give the right values for the UnProject matrix
248  GLfloat winX = (GLfloat(event->pos().x()) / GLfloat(width())) * 2 - 1;
249  GLfloat winY = -1 * ((GLfloat(event->pos().y()) / GLfloat(height())) * 2 - 1);
250  intersectModel(winX, winY);
251 
252  if (m_moveMarkerMode)
253  {
254  m_moveMarkerMode = false;
255  if (event->modifiers() == Qt::NoModifier) emit EmitMoveMarkerRequest(QString::fromStdString(m_moveMarkerName), m_cursor3DPosition);
256  return;
257  }
258 
259  while (true)
260  {
261  if (event->buttons() & Qt::LeftButton && event->modifiers() == Qt::NoModifier)
262  {
263  int trackballRadius;
264  if (width() < height()) trackballRadius = int(float(width()) / 2.2f);
265  else trackballRadius = int(float(height()) / 2.2f);
266  m_trackballStartCameraVec = QVector3D(m_cameraVecX, m_cameraVecY, m_cameraVecZ);
267  m_trackballStartUp = QVector3D(m_upX, m_upY, m_upZ);
268  m_trackball->StartTrackball(event->pos().x(), event->pos().y(), width() / 2, height() / 2, trackballRadius,
269  pgd::Vector3(double(m_trackballStartUp.x()), double(m_trackballStartUp.y()), double(m_trackballStartUp.z())),
270  pgd::Vector3(double(-m_trackballStartCameraVec.x()), double(-m_trackballStartCameraVec.y()), double(-m_trackballStartCameraVec.z())));
271  m_trackballFlag = true;
272  emit EmitStatusString(tr("Rotate"), 2);
273  update();
274  break;
275  }
276  if (event->buttons() & Qt::LeftButton && event->modifiers() & Qt::ShiftModifier)
277  {
278  // detect the collision point of the mouse click
279  if (m_hits.size() > 0)
280  {
281  auto closestHit = getClosestHit();
282  m_cursor3DPosition = QVector3D(float(closestHit->worldLocation().x), float(closestHit->worldLocation().y), float(closestHit->worldLocation().z));
283  QClipboard *clipboard = QApplication::clipboard();
284  clipboard->setText(QString("%1\t%2\t%3").arg(double(m_cursor3DPosition.x())).arg(double(m_cursor3DPosition.y())).arg(double(m_cursor3DPosition.z())), QClipboard::Clipboard);
285  emit EmitStatusString(QString("3D Cursor %1\t%2\t%3").arg(double(m_cursor3DPosition.x())).arg(double(m_cursor3DPosition.y())).arg(double(m_cursor3DPosition.z())), 2);
286  update();
287  }
288  break;
289  }
290 
291  if ((event->buttons() & Qt::MiddleButton && event->modifiers() & Qt::NoModifier) || (event->buttons() & Qt::LeftButton && event->modifiers() & Qt::AltModifier))
292  {
293  m_panStartCOI = QVector3D(m_COIx, m_COIy, m_COIz);
294  m_projectPanMatrix = m_proj * m_view; // model would be identity so mvpMatrix isn't needed
295  bool invertible;
296  m_unprojectPanMatrix = m_projectPanMatrix.inverted(&invertible); // we need the unproject matrix
297  if (!invertible)
298  {
299  qDebug() << "Problem inverting (m_proj * m_view)";
300  return;
301  }
302  m_panFlag = true;
303  // detect the collision point of the mouse click
304  if (m_hits.size() > 0)
305  {
306  auto closestHit = getClosestHit();
307  m_panStartPoint = QVector3D(float(closestHit->worldLocation().x), float(closestHit->worldLocation().y), float(closestHit->worldLocation().z));
308  QVector3D screenStartPoint = m_projectPanMatrix.map(m_panStartPoint);
309  m_panStartScreenPoint.setZ(screenStartPoint.z());
310  }
311  else // this is harder since we don't know the screen Z were are interested in.
312  {
313  // generate a screen Z from projecting the COI into screen coordinates (-1 to 1 box)
314  QVector3D screenStartPoint = m_projectPanMatrix.map(m_panStartCOI);
315  m_panStartScreenPoint.setZ(screenStartPoint.z());
316  // now unproject this point to get the pan start point
317  m_panStartPoint = m_unprojectPanMatrix.map(m_panStartScreenPoint);
318  }
319  emit EmitStatusString(tr("Pan"), 2);
320  update();
321  break;
322  }
323  if (event->buttons() & Qt::MiddleButton && event->modifiers() & Qt::AltModifier)
324  {
325  if (m_hits.size() > 0)
326  {
327  auto closestHit = getClosestHit();
328  QVector3D worldIntersection = QVector3D(float(closestHit->worldLocation().x), float(closestHit->worldLocation().y), float(closestHit->worldLocation().z));
329  m_COIx = worldIntersection.x();
330  m_COIy = worldIntersection.y();
331  m_COIz = worldIntersection.z();
332  QClipboard *clipboard = QApplication::clipboard();
333  clipboard->setText(QString("%1\t%2\t%3").arg(double(worldIntersection.x())).arg(double(worldIntersection.y())).arg(double(worldIntersection.z())), QClipboard::Clipboard);
334  emit EmitStatusString(QString("Centre of Interest %1\t%2\t%3").arg(double(worldIntersection.x())).arg(double(worldIntersection.y())).arg(double(worldIntersection.z())), 2);
335  emit EmitCOI(worldIntersection.x(), worldIntersection.y(), worldIntersection.z());
336  update();
337  }
338  break;
339  }
340  if ((event->buttons() & Qt::RightButton && event->modifiers() == Qt::NoModifier) || (event->buttons() & Qt::LeftButton && event->modifiers() & Qt::ControlModifier))
341  {
342  menuRequest(event->pos());
343  break;
344  }
345  break;
346  }

349 void SimulationWidget::mouseMoveEvent(QMouseEvent *event)
350 {
351  // qDebug() << event->localPos();
352  if (m_moveMarkerMode)
353  {
354  if (event->modifiers() == Qt::NoModifier)
355  {
356  GLfloat winX = (GLfloat(event->pos().x()) / GLfloat(width())) * 2 - 1;
357  GLfloat winY = -1 * ((GLfloat(event->pos().y()) / GLfloat(height())) * 2 - 1);
358  intersectModel(winX, winY);
359  auto closestHit = getClosestHit();
360  if (!closestHit) return;
361  m_cursor3DPosition = QVector3D(float(closestHit->worldLocation().x), float(closestHit->worldLocation().y), float(closestHit->worldLocation().z));
362  update();
363  return;
364  }
365  else
366  {
367  m_moveMarkerMode = false;
368  }
369  return;
370  }
371  while (true)
372  {
373  if (event->buttons() & Qt::LeftButton && event->modifiers() == Qt::NoModifier)
374  {
375  if (m_trackballFlag)
376  {
377  pgd::Quaternion pgdRotation;
378  m_trackball->RollTrackballToClick(event->pos().x(), event->pos().y(), &pgdRotation);
379  QQuaternion rotation(float(pgdRotation.n), float(pgdRotation.x), float(pgdRotation.y), float(pgdRotation.z));
380  rotation = rotation.conjugated();
381  QVector3D newCameraVec = rotation * m_trackballStartCameraVec;
382  m_cameraVecX = newCameraVec.x();
383  m_cameraVecY = newCameraVec.y();
384  m_cameraVecZ = newCameraVec.z();
385  QVector3D newUp = rotation * m_trackballStartUp;
386  m_upX = newUp.x();
387  m_upY = newUp.y();
388  m_upZ = newUp.z();
389  update();
390  emit EmitStatusString(QString("Camera %1 %2 %3 Up %4 %5 %6").arg(double(m_cameraVecX)).arg(double(m_cameraVecY)).arg(double(m_cameraVecZ)).arg(double(m_upX)).arg(double(m_upY)).arg(double(m_upZ)), 2);
391  }
392  break;
393  }
394  if ((event->buttons() & Qt::MiddleButton && event->modifiers() == Qt::NoModifier) || (event->buttons() & Qt::LeftButton && event->modifiers() & Qt::AltModifier))
395  {
396  if (m_panFlag)
397  {
398  GLfloat winX = (GLfloat(event->pos().x()) / GLfloat(width())) * 2 - 1;
399  GLfloat winY = -1 * ((GLfloat(event->pos().y()) / GLfloat(height())) * 2 - 1);
400  QVector3D screenPoint(winX, winY, m_panStartScreenPoint.z());
401  QVector3D panCurrentPoint = m_unprojectPanMatrix.map(screenPoint);
402  m_COIx = m_panStartCOI.x() - (panCurrentPoint.x() - m_panStartPoint.x());
403  m_COIy = m_panStartCOI.y() - (panCurrentPoint.y() - m_panStartPoint.y());
404  m_COIz = m_panStartCOI.z() - (panCurrentPoint.z() - m_panStartPoint.z());
405  // qDebug() << "panCurrentPoint=" << panCurrentPoint;
406  emit EmitStatusString(QString("COI %1 %2 %3").arg(double(m_COIx)).arg(double(m_COIy)).arg(double(m_COIz)), 2);
407  emit EmitCOI(m_COIx, m_COIy, m_COIz);
408  update();
409  }
410  break;
411  }
412  break;
413  }

438 void SimulationWidget::keyPressEvent(QKeyEvent *event)
439 {
440  if (m_moveMarkerMode) m_moveMarkerMode = false;
441  QVector3D newPosition = m_cursor3DPosition;
442  switch ( event->key() )
443  {
444 
445  // X, Y and Z move the cursor
446  case Qt::Key_X:
447  if (event->modifiers() == Qt::NoModifier)
448  newPosition.setX(newPosition.x() + m_cursor3DNudge);
449  else
450  newPosition.setX(newPosition.x() - m_cursor3DNudge);
451  break;
452 
453  case Qt::Key_Y:
454  if (event->modifiers() == Qt::NoModifier)
455  newPosition.setY(newPosition.y() + m_cursor3DNudge);
456  else
457  newPosition.setY(newPosition.y() - m_cursor3DNudge);
458  break;
459 
460  case Qt::Key_Z:
461  if (event->modifiers() == Qt::NoModifier)
462  newPosition.setZ(newPosition.z() + m_cursor3DNudge);
463  else
464  newPosition.setZ(newPosition.z() - m_cursor3DNudge);
465  break;
466 
467  // S snaps the cursor to the nearest whole number multiple of the nudge value
468  case Qt::Key_S:
469  newPosition.setX(round(newPosition.x() / m_cursor3DNudge) * m_cursor3DNudge);
470  newPosition.setY(round(newPosition.y() / m_cursor3DNudge) * m_cursor3DNudge);
471  newPosition.setZ(round(newPosition.z() / m_cursor3DNudge) * m_cursor3DNudge);
472  break;
473  }
474 
475  if (newPosition != m_cursor3DPosition)
476  {
477  m_cursor3DPosition = newPosition;
478  QClipboard *clipboard = QApplication::clipboard();
479  clipboard->setText(QString("%1\t%2\t%3").arg(double(newPosition.x())).arg(double(newPosition.y())).arg(double(newPosition.z())), QClipboard::Clipboard);
480  emit EmitStatusString(QString("3D Cursor %1\t%2\t%3").arg(double(newPosition.x())).arg(double(newPosition.y())).arg(double(newPosition.z())), 2);
481  update();
482  }

1077 bool SimulationWidget::intersectModel(float winX, float winY)
1078 {
1079  m_hits.clear();
1080  m_hitsIndexByZ.clear();
1081  std::vector<pgd::Vector3> intersectionCoordList;
1082  std::vector<size_t> intersectionIndexList;
1083  bool hit;
1084  for (auto &&drawableIter : m_drawables)
1085  {
1086  for (auto &&facetedObjectIter : drawableIter->facetedObjectList())
1087  {
1088  QMatrix4x4 mvpMatrix = m_proj * m_view * facetedObjectIter->model();
1089  bool invertible;
1090  QMatrix4x4 unprojectMatrix = mvpMatrix.inverted(&invertible);
1091  if (!invertible) // usually because the scale is zero so not an error condition
1092  {
1093  qDebug() << "mvpMatrix matrix not invertible: " << drawableIter->name().c_str();
1094  qDebug() << "m_proj " << m_proj;
1095  qDebug() << "m_view " << m_view;
1096  qDebug() << "facetedObjectIter->model() " << facetedObjectIter->model();
1097  break;
1098  }
1099  QVector4D screenPoint(winX, winY, -1, 1);
1100  QVector4D nearPoint4D = unprojectMatrix * screenPoint;
1101  screenPoint.setZ(+1);
1102  QVector4D farPoint4D = unprojectMatrix * screenPoint;
1103  QVector3D rayOrigin = nearPoint4D.toVector3DAffine();
1104  QVector3D rayVector = farPoint4D.toVector3DAffine() - rayOrigin;
1105  pgd::Vector3 origin(double(rayOrigin.x()), double(rayOrigin.y()), double(rayOrigin.z()));
1106  pgd::Vector3 vector(double(rayVector.x()), double(rayVector.y()), double(rayVector.z()));
1107  pgd::Vector3 vectorNorm = vector / vector.Magnitude();
1108 
1109  intersectionCoordList.clear();
1110  intersectionIndexList.clear();
1111  hit = facetedObjectIter->FindIntersection(origin, vectorNorm, &intersectionCoordList, &intersectionIndexList);
1112  if (hit)
1113  {
1114  for (size_t i = 0; i < intersectionCoordList.size(); i++)
1115  {
1116  auto newHits = std::make_unique<IntersectionHits>();
1117  newHits->setDrawable(drawableIter);
1118  newHits->setFacetedObject(facetedObjectIter);
1119  newHits->setTriangleIndex(intersectionIndexList[i]);
1120  newHits->setModelLocation(intersectionCoordList[i]);
1121  QVector3D modelIntersection(float(intersectionCoordList[i].x), float(intersectionCoordList[i].y), float(intersectionCoordList[i].z));
1122  QVector3D screenIntersection = mvpMatrix.map(modelIntersection);
1123  if (screenIntersection.z() < -1.0f || screenIntersection.z() > 1.0f) continue; // this means that only visible intersections are allowed
1124  QVector3D worldIntersection = facetedObjectIter->model().map(modelIntersection);
1125  newHits->setWorldLocation(pgd::Vector3(double(worldIntersection.x()), double(worldIntersection.y()), double(worldIntersection.z())));
