139 std::string *Simulation::LoadModel(const char *buffer, size_t length) // note this requires buffer to be a 0 terminated string of size length + 1
140 {
141  std::string *ptr = m_parseXML.LoadModel(buffer, length, "GAITSYM2019"s);
142  if (ptr) return ptr;
143 
144  // this logic allows forward references at the expense of slightly less obvious error messages
145  std::list<ParseXML::XMLElement *> unprocessedList;
146  for (auto &&it : *m_parseXML.elementList()) unprocessedList.push_back(it.get());
147  size_t lastSize = 0;
148  size_t cycles = 0;
149  std::vector<std::string> errorList;
150  while (unprocessedList.size() > 0 && unprocessedList.size() != lastSize)
151  {
152  cycles++;
153  lastSize = unprocessedList.size();
154  errorList.clear();
155  for (auto it = unprocessedList.begin(); it != unprocessedList.end();)
156  {
157  lastErrorPtr()->clear();
158  if ((*it)->tag == "GLOBAL"s) ParseGlobal(*it);
159  else if ((*it)->tag == "BODY"s) ParseBody(*it);
160  else if ((*it)->tag == "JOINT"s) ParseJoint(*it);
161  else if ((*it)->tag == "GEOM"s) ParseGeom(*it);
162  else if ((*it)->tag == "STRAP"s) ParseStrap(*it);
163  else if ((*it)->tag == "MUSCLE"s) ParseMuscle(*it);
164  else if ((*it)->tag == "DRIVER"s) ParseDriver(*it);
165  else if ((*it)->tag == "DATATARGET"s) ParseDataTarget(*it);
166  else if ((*it)->tag == "MARKER"s) ParseMarker(*it);
167  else if ((*it)->tag == "REPORTER"s) ParseReporter(*it);
168  else if ((*it)->tag == "CONTROLLER"s) ParseController(*it);
169  else if ((*it)->tag == "WAREHOUSE"s) ParseWarehouse(*it);
170  else if ((*it)->tag == "FLUIDSAC"s) ParseFluidSac(*it);
171  if (lastErrorPtr()->size())
172  {
173  errorList.push_back(*lastErrorPtr());
174  it++;
175  }
176  else
177  {
178  it = unprocessedList.erase(it);
179  }
180  }
181  }
182  if (lastErrorPtr()->size())
183  {
184  setLastError(pystring::join("\n"s, errorList));
185  return lastErrorPtr();
186  }
187  if (cycles > 1)
188  std::cerr << "Warning: file took " << cycles << " cycles to parse. Consider reordering for speed.\n";
189 
190  // joints are created with the bodies in construction poses
191  // then the bodies are moved to their starting poses
192  for (auto &&it : m_BodyList) it.second->LateInitialisation();
193  // and we recalculate the dynamic items with the new muscle positions
194  for (auto &&it : m_MuscleList) it.second->LateInitialisation();
195  for (auto &&it : m_FluidSacList) it.second->LateInitialisation();
196  // and some joints require things to be done after the bodies are moved to their start positions
197  for (auto &&it : m_JointList) it.second->LateInitialisation();
198 
199  // for the time being just set the current warehouse to the first one in the list
200  if (m_global->CurrentWarehouseFile().length() == 0 && m_WarehouseList.size() > 0) m_global->setCurrentWarehouseFile(m_WarehouseList.begin()->first);
201 
202  // and we need to set the cycle time
203  // currently just using the maximum value but some sort of fuzzy lowest common multiple might be better
204  // the easiest way to do that is to mutiply by an appropriate power of 10 with nearest number rounding (int(v * 10000 + 0.5)) to make the numbers into integers and then use an integer formula and convert back
205  // using std::lcm from numeric with accumulate so it works on a container (a std::set makes sense for longer lists perhas)
206  // std::vector<int> v{4, 6, 10};
207  // auto lcm = std::accumulate(v.begin(), v.end(), 1, [](auto & a, auto & b) { return std::lcm(a, b); });
208  m_CycleTime = 0;
209  for (auto &&driver : m_DriverList)
210  {
211  CyclicDriver *cyclicDriver = dynamic_cast<CyclicDriver*>(driver.second.get());
212  if (cyclicDriver) m_CycleTime = std::max(cyclicDriver->GetCycleTime(), m_CycleTime);
213  StackedBoxcarDriver *stackedBoxcarDriver = dynamic_cast<StackedBoxcarDriver*>(driver.second.get());
214  if (stackedBoxcarDriver) m_CycleTime = std::max(stackedBoxcarDriver->GetCycleTime(), m_CycleTime);
215  }
216 #ifdef OUTPUTS_AFTER_SIMULATION_STEP
217  if (m_OutputModelStateAtTime == 0.0 || m_OutputModelStateAtCycle == 0)
218  {
219  OutputProgramState();
220  m_OutputModelStateAtTime = -1.0;
221  m_OutputModelStateAtCycle = -1;
222  }
223 #endif
224 
225  return nullptr;
226 }

229 //----------------------------------------------------------------------------
230 void Simulation::UpdateSimulation()
231 {
232  // calculate the warehouse and position matching fitnesses before we move to a new location
233  while (true)
234  {
235 #ifdef EXPERIMENTAL
236  if (m_global->fitnessType() == Global::ClosestWarehouse)
237  {
238  auto warehouseIter = m_WarehouseList.find(m_global->CurrentWarehouseFile());
239  if (warehouseIter != m_WarehouseList.end())
240  {
241  WarehouseUnit *warehouseUnit = warehouseIter->second->GetWarehouseUnit(0); // only interested in the warehouse unit 0 for this measurement
242  warehouseUnit->SetBodyQueryData(m_BodyList);
243  warehouseUnit->DoSearch();
244  m_WarehouseDistance = warehouseUnit->GetNearestNeighbourDistance();
245 #ifdef TOTAL_DISTANCE_WAREHOUSE_METRIC // this version calculates a total distance from warehouse metric
246  if (m_WarehouseDistance < m_WarehouseUnitIncreaseDistanceThreshold) m_ClosestWarehouseFitness += (m_WarehouseUnitIncreaseDistanceThreshold - m_WarehouseDistance);
247 #else // and this is a minimum distance metric
248  if (m_WarehouseDistance < std::fabs(m_ClosestWarehouseFitness))
249  {
250  m_ClosestWarehouseFitness = -m_WarehouseDistance; // negative because we always try and maximise the value
251  std::cerr << m_SimulationTime << " m_ClosestWarehouseFitness=" << m_ClosestWarehouseFitness << "\n";
252  }
253 #endif
254  }
255  break;
256  }
257 #endif
258  if (m_global->fitnessType() == Global::KinematicMatch || m_global->fitnessType() == Global::KinematicMatchMiniMax)
259  {
260  double minScore = DBL_MAX;
261  for (auto &&it : m_DataTargetList)
262  {
263  double matchScore;
264  bool matchScoreValid;
265  std::tie(matchScore, matchScoreValid) = it.second->calculateMatchValue(m_SimulationTime);
266  if (matchScoreValid)
267  {
268  m_KinematicMatchFitness += matchScore;
269  if (matchScore < minScore)
270  minScore = matchScore;
271  }
272  }
273  if (minScore < DBL_MAX)
274  m_KinematicMatchMiniMaxFitness += minScore;
275  break;
276  }
277 
278  break;
279  }
280 
281  // now start the actual simulation
282 
283  // check collisions first
284  dJointGroupEmpty(m_ContactGroup);
285  m_ContactList.clear();
286  for (auto &&geomIter : m_GeomList) geomIter.second->ClearContacts();
287  dSpaceCollide(m_SpaceID, this, &NearCallback);
288 
289 #ifdef EXPERIMENTAL
290  auto warehouseIter = m_WarehouseList.find(m_global->CurrentWarehouseFile());
291  if (warehouseIter != m_WarehouseList.end() && m_global->fitnessType() != Global::ClosestWarehouse)
292  {
293  warehouseIter->second->DoSearch(m_BodyList);
294  m_WarehouseDistance = warehouseIter->second->GetNearestNeighbourDistance();
295  std::vector<std::string> *driverNames = warehouseIter->second->GetDriverNames();
296  double *activations = warehouseIter->second->GetCurrentActivations();
297  for (unsigned int iDriver = 0; iDriver < driverNames->size(); iDriver++)
298  {
299  FixedDriver *driver = dynamic_cast<FixedDriver *>(m_DriverList[driverNames->at(iDriver)].get());
300  if (driver)
301  {
302  driver->setValue(activations[iDriver]);
303  }
304  else
305  {
306  std::cerr << "Only FixedDriver currently supported as warehouse targets\n";
307  }
308  }
309  std::cerr << m_SimulationTime << " m_WarehouseDistance=" << m_WarehouseDistance << "\n";
310  }
311 #endif
312 
313  // update the drivers
314  for (auto &&it : m_DriverList)
315  {
316  it.second->Update();
317  it.second->SendData();
318  }
319  // and the controllers (which are drivers too probably)
320  for (auto &&it : m_ControllerList)
321  {
322  auto driver = dynamic_cast<Driver *>(it.second.get());
323  if (driver)
324  {
325  driver->Update();
326  driver->SendData();
327  }
328  if (it.second->lastStepCount() != m_StepCount)
329  std::cerr << "Warning: " << it.first << " controller not updated\n"; // currently cannot stack controllers although this is fixable
330  }
331 
332  // update the muscles
333  for (auto iter1 = m_MuscleList.begin(); iter1 != m_MuscleList.end(); /* no increment */)
334  {
335  iter1->second->CalculateStrap();
336  iter1->second->SetActivation();
337 
338  // check for breaking strain
339  DampedSpringMuscle *dampedSpringMuscle = dynamic_cast<DampedSpringMuscle *>(iter1->second.get());
340  if (dampedSpringMuscle)
341  {
342  if (dampedSpringMuscle->ShouldBreak())
343  {
344  iter1 = m_MuscleList.erase(iter1); // erase returns the next iterator [but m_MuscleList.erase(iter1++) would also work and is compatible with older C++ compilers]
345  continue;
346  }
347  }
348 
349  std::vector<std::unique_ptr<PointForce>> *pointForceList = iter1->second->GetPointForceList();
350  double tension = iter1->second->GetTension();
351 #ifdef DEBUG_CHECK_FORCES
352  pgd::Vector3 force(0, 0, 0);
353 #endif
354  for (unsigned int i = 0; i < pointForceList->size(); i++)
355  {
356  PointForce *pointForce = (*pointForceList)[i].get();
357  if (pointForce->body)
358  dBodyAddForceAtPos(pointForce->body->GetBodyID(),
359  pointForce->vector[0] * tension, pointForce->vector[1] * tension, pointForce->vector[2] * tension,
360  pointForce->point[0], pointForce->point[1], pointForce->point[2]);
361 #ifdef DEBUG_CHECK_FORCES
362  force += pgd::Vector3(pointForce->vector[0] * tension, pointForce->vector[1] * tension, pointForce->vector[2] * tension);
363 #endif
364  }
365 #ifdef DEBUG_CHECK_FORCES
366  std::cerr.setf(std::ios::floatfield, std::ios::fixed);
367  std::cerr << iter1->first << " " << force.x << " " << force.y << " " << force.z << "\n";
368  std::cerr.unsetf(std::ios::floatfield);
369 #endif
370  iter1++; // this has to be done outside the for definition because erase returns the next iterator
371  }
372 
373  // update the joints (needed for motors, end stops and stress calculations)
374  for (auto &&jointIter : m_JointList) jointIter.second->Update();
375 
376  // update the fluid sacs
377  for (auto fsIter = m_FluidSacList.begin(); fsIter != m_FluidSacList.end(); fsIter++)
378  {
379  fsIter->second->calculateVolume();
380  fsIter->second->calculatePressure();
381  fsIter->second->calculateLoadsOnMarkers();
382  for (size_t i = 0; i < fsIter->second->pointForceList().size(); i++)
383  {
384  const PointForce *pf = &fsIter->second->pointForceList().at(i);
385  dBodyAddForceAtPos(pf->body->GetBodyID(), pf->vector[0], pf->vector[1], pf->vector[2], pf->point[0], pf->point[1], pf->point[2]);
386  }
387  }
388 
389 #ifdef EXPERIMENTAL
390  // update the bodies (needed for drag calculations)
391  for (auto &&bodyIter : m_BodyList) bodyIter.second->ComputeDrag();
392 #endif
393 
394 #ifndef OUTPUTS_AFTER_SIMULATION_STEP
395  if (m_OutputWarehouseFlag) OutputWarehouse();
396  if (m_OutputModelStateAtTime >= 0.0)
397  {
398  if (m_SimulationTime >= m_OutputModelStateAtTime)
399  {
400  OutputProgramState();
401  m_OutputModelStateAtTime = -1;
402  }
403  }
404  else if (m_OutputModelStateAtCycle >= 0 && m_CycleTime >= 0 && m_SimulationTime >= m_CycleTime * m_OutputModelStateAtCycle)
405  {
406  OutputProgramState();
407  m_OutputModelStateAtCycle = -1;
408  }
409  else if (m_OutputModelStateAtWarehouseDistance > 0 && m_WarehouseDistance >= m_OutputModelStateAtWarehouseDistance)
410  {
411  OutputProgramState();
412  m_OutputModelStateAtWarehouseDistance = 0;
413  }
414 #endif
415 
416  // run the simulation
417  switch (m_global->stepType())
418  {
419  case Global::World:
420  dWorldStep(m_WorldID, m_global->StepSize());
421  break;
422 
423  case Global::Quick:
424  dWorldQuickStep(m_WorldID, m_global->StepSize());
425  break;
426  }
427 
428  // test for penalties
429  if (m_errorHandler.IsMessage()) m_KinematicMatchFitness += m_global->NumericalErrorsScore();
430 
431  // calculate the energies
432  for (auto &&iter1 : m_MuscleList)
433  {
434  m_MechanicalEnergy += iter1.second->GetPower() * m_global->StepSize();
435  m_MetabolicEnergy += iter1.second->GetMetabolicPower() * m_global->StepSize();
436  }
437  m_MetabolicEnergy += m_global->BMR() * m_global->StepSize();
438 
439  // update any contact force dependent drivers (because only after the simulation is the force valid
440  // update the footprint indicator
441  if (m_ContactList.size() > 0)
442  {
443  for (auto &&it : m_DriverList)
444  {
445  TegotaeDriver *tegotaeDriver = dynamic_cast<TegotaeDriver *>(it.second.get());
446  if (tegotaeDriver) tegotaeDriver->UpdateReactionForce();
447  }
448  }
449 
450  // all reporting is done after a simulation step
451 
452  DumpObjects();
453 
454  // update the time counter
455  m_SimulationTime += m_global->StepSize();
456 
457  // update the step counter
458  m_StepCount++;
459 
460 
461 #ifdef OUTPUTS_AFTER_SIMULATION_STEP
462 // if (m_OutputKinematicsFlag && m_StepCount % gDisplaySkip == 0) OutputKinematics();
463  if (m_OutputWarehouseFlag) OutputWarehouse();
464  if (m_OutputModelStateAtTime > 0.0)
465  {
466  if (m_SimulationTime >= m_OutputModelStateAtTime)
467  {
468  OutputProgramState();
469  m_OutputModelStateAtTime = 0.0;
470  }
471  }
472  else if (m_OutputModelStateAtCycle >= 0 && m_CycleTime >= 0 && m_SimulationTime >= m_CycleTime * m_OutputModelStateAtCycle)
473  {
474  OutputProgramState();
475  m_OutputModelStateAtCycle = -1;
476  }
477  else if (m_OutputModelStateAtWarehouseDistance > 0 && m_WarehouseDistance >= m_OutputModelStateAtWarehouseDistance)
478  {
479  OutputProgramState();
480  m_OutputModelStateAtWarehouseDistance = 0;
481  }
482 #endif
483 }

485 //----------------------------------------------------------------------------
486 bool Simulation::TestForCatastrophy()
487 {
488  // first of all check to see that ODE is happy
489  if (m_errorHandler.IsMessage())
490  {
491  int num = m_errorHandler.GetLastMessageNumber();
492  std::string messageText = m_errorHandler.GetLastMessage();
493  m_numericalErrorCount++;
494  if (m_global->PermittedNumericalErrors() >= 0 && m_numericalErrorCount > m_global->PermittedNumericalErrors())
495  {
496  std::cerr << "t=" << m_SimulationTime << " error count=" << m_numericalErrorCount << " Failed due to ODE warning " << num << " " << messageText << "\n";
497  return true;
498  }
499  else
500  {
501  std::cerr << "t=" << m_SimulationTime << " ODE warning " << num << " " << messageText << "\n";
502  }
503  m_errorHandler.ClearMessage();
504  }
505 
506  // check for simulation error
507  if (m_SimulationError)
508  {
509  std::cerr << "Failed due to simulation error " << m_SimulationError << "\n";
510  return true;
511  }
512 
513  // check for contact abort
514  if (m_ContactAbort)
515  {
516  std::cerr << "Failed due to contact abort\n";
517  for (auto &&it: m_ContactAbortList) { std::cerr << it << "\n"; }
518  return true;
519  }
520 
521  // check for data target abort
522  if (m_DataTargetAbort)
523  {
524  std::cerr << "Failed due to DataTarget abort\n";
525  for (auto &&it: m_DataTargetAbortList) { std::cerr << it << "\n"; }
526  return true;
527  }
528 
529  // check that all bodies meet velocity and stop conditions
530 
531  Body::LimitTestResult p;
532  for (auto &&iter1 : m_BodyList)
533  {
534  p = iter1.second->TestLimits();
535  switch (p)
536  {
537  case Body::WithinLimits:
538  break;
539 
540  case Body::XPosError:
541  case Body::YPosError:
542  case Body::ZPosError:
543  std::cerr << "Failed due to position error " << Body::limitTestResultStrings(p) << " in: " << iter1.second->name() << "\n";
544  return true;
545 
546  case Body::XVelError:
547  case Body::YVelError:
548  case Body::ZVelError:
549  std::cerr << "Failed due to linear velocity error " << Body::limitTestResultStrings(p) << " in: " << iter1.second->name() << "\n";
550  return true;
551 
552  case Body::XAVelError:
553  case Body::YAVelError:
554  case Body::ZAVelError:
555  std::cerr << "Failed due to angular velocity error " << Body::limitTestResultStrings(p) << " in: " << iter1.second->name() << "\n";
556  return true;
557 
558  case Body::NumericalError:
559  std::cerr << "Failed due to numerical error " << Body::limitTestResultStrings(p) << " in: " << iter1.second->name() << "\n";
560  return true;
561  }
562  }
563 
564  HingeJoint *j;
565  FixedJoint *f;
566  int t;
567  for (auto &&iter3 : m_JointList)
568  {
569  j = dynamic_cast<HingeJoint *>(iter3.second.get());
570  if (j)
571  {
572  t = j->TestLimits();
573  if (t < 0)
574  {
575  std::cerr << "Failed due to LoStopTorqueLimit error in: " << iter3.second->name() << "\n";
576  return true;
577  }
578  else if (t > 0)
579  {
580  std::cerr << "Failed due to HiStopTorqueLimit error in: " << iter3.second->name() << "\n";
581  return true;
582  }
583  }
584 
585  f = dynamic_cast<FixedJoint *>(iter3.second.get());
586  if (f)
587  {
588  if (f->CheckStressAbort())
589  {
590  std::cerr << "Failed due to stress limit error in: " << iter3.second->name() << " " << f->GetLowPassMinStress() << " " << f->GetLowPassMaxStress() << "\n";
591  return true;
592  }
593  }
594  }
595 
596  // and test the reporters for stop conditions
597  for (auto &&reporterIter : m_ReporterList)
598  {
599  if (reporterIter.second->ShouldAbort())
600  {
601  std::cerr << "Failed due to Reporter Abort in: " << reporterIter.second->name() << "\n";
602  return true;
603  }
604  }
605 
606 #ifdef EXPERIMENTAL
607  // test for WarehouseFailDistanceAbort if set

1447 void Simulation::NearCallback(void *data, dGeomID o1, dGeomID o2)
1448 {
1449  Simulation *s = reinterpret_cast<Simulation *>(data);
1450  Geom *g1 = reinterpret_cast<Geom *>(dGeomGetData(o1));
1451  Geom *g2 = reinterpret_cast<Geom *>(dGeomGetData(o2));
1452 
1453  dBodyID b1 = dGeomGetBody(o1);
1454  dBodyID b2 = dGeomGetBody(o2);
1455  if (b1 == b2)
1456  {
1457  return; // it is never useful for two contacts on the same body to collide [I'm not sure if this every happens - FIX ME - set up a test]
1458  }
1459 
1460  if (s->m_global->AllowConnectedCollisions() == false)
1461  {
1462  if (b1 && b2 && dAreConnectedExcluding(b1, b2, dJointTypeContact)) return;
1463  }
1464 
1465  if (s->m_global->AllowInternalCollisions() == false)
1466  {
1467  if (g1->GetGeomLocation() == g2->GetGeomLocation()) return;
1468  }
1469 
1470  if (g1->GetExcludeList()->size())
1471  {
1472  std::vector<Geom *> *excludeList = g1->GetExcludeList();
1473  for (size_t i = 0; i < excludeList->size(); i++)
1474  {
1475  if (excludeList->at(i) == g2) return;
1476  }
1477  }
1478  if (g2->GetExcludeList()->size())
1479  {
1480  std::vector<Geom *> *excludeList = g2->GetExcludeList();
1481  for (size_t i = 0; i < excludeList->size(); i++)
1482  {
1483  if (excludeList->at(i) == g1) return;
1484  }
1485  }
1486 
1487  std::unique_ptr<dContact[]> contact = std::make_unique<dContact[]>(size_t(s->m_MaxContacts));
1488  // the choice of std::max(cfm) and std::min(erp) means that the softest contact should be used
1489  double cfm = std::max(g1->GetContactSoftCFM(), g2->GetContactSoftCFM());
1490  double erp = std::min(g1->GetContactSoftERP(), g2->GetContactSoftERP());
1491  // just use the largest for mu, rho and bounce
1492  double mu = std::max(g1->GetContactMu(), g2->GetContactMu());
1493  double bounce = std::max(g1->GetContactBounce(), g2->GetContactBounce());
1494  double rho = std::max(g1->GetRho(), g2->GetRho());
1495  if (erp < 0) // the only one that needs checking because all the others are std::max so values <0 will never be chosen if one value is >0
1496  {
1497  if (g1->GetContactSoftERP() < 0) erp = g2->GetContactSoftERP();
1498  else erp = g1->GetContactSoftERP();
1499  }
1500  for (size_t i = 0; i < size_t(s->m_MaxContacts); i++)
1501  {
1502  contact[i].surface.mode = dContactApprox1;
1503  contact[i].surface.mu = mu;
1504  if (bounce >= 0)
1505  {
1506  contact[i].surface.bounce = bounce;
1507  contact[i].surface.mode += dContactBounce;
1508  }
1509  if (rho >= 0)
1510  {
1511  contact[i].surface.rho = rho;
1512  contact[i].surface.mode += dContactRolling;
1513  }
1514  if (cfm >= 0)
1515  {
1516  contact[i].surface.soft_cfm = cfm;
1517  contact[i].surface.mode += dContactSoftCFM;
1518  }
1519  if (erp >= 0)
1520  {
1521  contact[i].surface.soft_erp = erp;
1522  contact[i].surface.mode += dContactSoftERP;
1523  }
1524  }
1525  int numc = dCollide(o1, o2, s->m_MaxContacts, &contact[0].geom, sizeof(dContact));
1526  if (numc)
1527  {
1528  for (size_t i = 0; i < size_t(numc); i++)
1529  {
1530  if (g1->GetAbort()) s->SetContactAbort(g1->name());
1531  if (g2->GetAbort()) s->SetContactAbort(g2->name());
1532  dJointID c;
1533  if (g1->GetAdhesion() == false && g2->GetAdhesion() == false)
1534  {
1535  c = dJointCreateContact(s->m_WorldID, s->m_ContactGroup, &contact[i]);
1536  dJointAttach(c, b1, b2);
1537  std::unique_ptr<Contact> myContact = std::make_unique<Contact>();
1538  myContact->setSimulation(s);
1539  dJointSetFeedback(c, myContact->GetJointFeedback());
1540  myContact->SetJointID(c);
1541  std::copy_n(contact[i].geom.pos, dV3E__MAX, myContact->GetContactPosition());
1542 // // only add the contact information once
1543 // // and add it to the non-environment geom
1544 // if (g1->GetGeomLocation() == Geom::environment)
1545 // g2->AddContact(myContact.get());
1546 // else
1547 // g1->AddContact(myContact.get());
1548  // add the contact information to both geoms
1549  g1->AddContact(myContact.get());
1550  g2->AddContact(myContact.get());
1551  s->m_ContactList.push_back(std::move(myContact));

1553  else
1554  {
