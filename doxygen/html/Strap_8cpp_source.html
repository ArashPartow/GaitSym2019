31 std::string *Strap::createFromAttributes()
32 {
33  m_pointForceList.clear();
34  m_torqueMarkerList.clear();
35  if (NamedObject::createFromAttributes()) return lastErrorPtr();
36  std::string buf;
37  if (findAttribute("Length"s, &buf) == nullptr) return lastErrorPtr();
38  this->setLength(GSUtil::Double(buf));
39 
40  if (findAttribute("TorqueMarkerIDList"s, &buf))
41  {
42  std::vector<std::string> result;
43  pystring::split(buf, result);
44  if (result.size())
45  {
46  m_torqueMarkerList.reserve(result.size());
47  for (auto &&it: result)
48  {
49  auto torqueMarker = simulation()->GetMarkerList()->find(it);
50  if (torqueMarker == simulation()->GetMarkerList()->end())
51  {
52  setLastError("STRAP ID=\""s + name() +"\" torque marker \""s + it +"\" not found"s);
53  return lastErrorPtr();
54  }
55  m_torqueMarkerList.push_back(torqueMarker->second.get());
56  }
57  }
58  }
59  return nullptr;
60 }
