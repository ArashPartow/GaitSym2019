752  if a != 0:
753  point = [ d / a, 0.0, 0.0]
754  v1 = [-b / a, 1.0, 0.0]

792 def convert_to_erp_cfm(spring_constant, damping_constant, integration_stepsize):
793  h = integration_stepsize
794  kp = spring_constant
795  kd = damping_constant
796  erp = (h * kp) / ((h * kp) + kd)
797  cfm = 1.0 / ((h * kp) + kd)
798  return (erp, cfm)

800 def convert_to_spring_and_damping_constants(erp, cfm, integration_stepsize):
801  h = integration_stepsize
802  kp = erp / (cfm * h)
803  kd = (1.0 - erp) / cfm
804  spring_constant = kp
805  damping_constant = kd
806  return (spring_constant, damping_constant)
