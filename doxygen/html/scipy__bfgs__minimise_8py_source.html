179 def parse_insert(insert_string, genes, verbose):
180  # this routine is a little fragile because exprtk and python do not use quite the same syntax
181  # we try to fix things up in a rather dumb way that won't work in all cases
182  # most of the math functions come from the "from math import *" line so that the
183  # python eval statement can do all the work
184  # but I could probabl;y do something cleverer by spcifying globals and locals to the
185  # eval statement (but the globals() would be needed for all the math functions)
186 
187  # first step is to substitute the genes
188  match = re.search(r'g *\‍(', insert_string)
189  if match:
190  closing_bracket = find_unmatched_close_bracket(insert_string[match.end():])
191  if closing_bracket == -1:
192  print('Error parsing genes in %s' % (insert_string))
193  sys.exit(1)
194  index = int(0.5 + parse_insert(insert_string[match.end(): match.end() + closing_bracket], genes, verbose))
195  prefix = insert_string[:match.start()]
196  if match.end() + closing_bracket + 1 < len(insert_string):
197  suffix = insert_string[match.end() + closing_bracket + 1:]
198  else:
199  suffix = ''
200  return parse_insert('%s%.18e%s' % (prefix, genes[index], suffix), genes, verbose)
201 
202  # second step is to handle functions not understood by python
203  # if(test,result_when_true,result_when_false)
204  match = re.search(r'if *\‍(', insert_string)
205  if match:
206  closing_bracket = find_unmatched_close_bracket(insert_string[match.end():])
207  if closing_bracket == -1:
208  print('Error parsing genes in %s' % (insert_string))
209  sys.exit(1)
210  argument_string = insert_string[match.end(): match.end() + closing_bracket]
211  arguments = split_on_unbracketed_command(argument_string)
212  if len(arguments) != 3:
213  print('if() requires 3 arguments %s' % (insert_string))
214  sys.exit(1)
215  test_value = parse_insert(arguments[0], genes, verbose)
216  if test_value == 0:
217  result = arguments[2]
218  else:
219  result = arguments[1]
220  prefix = insert_string[:match.start()]
221  if match.end() + closing_bracket + 1 < len(insert_string):
222  suffix = insert_string[match.end() + closing_bracket + 1:]
223  else:
224  suffix = ''
225  return s('%s%s%s'% (prefix, result, suffix), genes, verbose)
226 
227  # exprtk uses ^ when python uses **
228  insert_string = insert_string.replace('^', '**')
229 
230  # we could now handle the other operators in order of precidence unaries, **^, */%, +-, booleans
231  # but there are a few gotchas (e.g. identifying unary operators) and python can now handle this
232  if verbose: print('eval("%s")' % (insert_string))

268 def create_smart_substitution(data):
269  smart_substitution_text_components = []
270  smart_substitution_parser_text = []
271  ptr1 = 0
272  ptr2 = data.find("[[")
273  if ptr2 == -1:
274  print("Error: could not find any [[\n")
275  sys.exit(1)
276  while True:
277  # print(ptr1, ptr2)
278  smart_substitution_text_components.append(data[ptr1: ptr2])
279  ptr2 += 2
280  ptr1 = data[ptr2:].find("]]")
281  if ptr1 == -1:
282  print("Error: could not find matching ]]\n")
283  sys.exit(1)
284  ptr1 += ptr2
285  smart_substitution_parser_text.append(data[ptr2: ptr1])
286  ptr1 += 2
287  ptr2 = data[ptr1:].find("[[")
288  if ptr2 == -1:
289  break
290  ptr2 += ptr1
291  smart_substitution_text_components.append(data[ptr1:])
292  for i in range(0, len(smart_substitution_parser_text)): # convert g[] to g()
293  smart_substitution_parser_text[i] = re.sub(r'g\[(\d+)\]', r'g(\1)', smart_substitution_parser_text[i])
294  return (smart_substitution_text_components, smart_substitution_parser_text)

296 def read_ranges(file_name):
297  ranges = {}
298  with open(file_name, 'r', encoding='utf8') as f_in:
299  lines = f_in.read().splitlines()
300  if len(lines) == 0:
301  print('Error: could not parse "%s"' % (file_name))
302  sys.exit(1)
303  column_names = lines[0].split()
304  if len(column_names) == 0:
305  print('Error: could not parse column_names in "%s"' % (file_name))
306  sys.exit(1)
307  for column_name in column_names:
308  ranges[column_name] = []
309 
310  for i in range(1, len(lines)):
311  tokens = lines[i].split()
312  if len(tokens) == 0: # skip blank lines
313  continue
314  if len(tokens) != len(column_names):
315  print('Error: could not parse data in "%s"' % (file_name))
316  sys.exit(1)
317  for j in range(0, len(tokens)):
318  ranges[column_names[j]].append(float(tokens[j]))
319  return ranges
