175 def parse_insert(insert_string, genes, verbose):
176  # this routine is a little fragile because exprtk and python do not use quite the same syntax
177  # we try to fix things up in a rather dumb way that won't work in all cases
178  # most of the math functions come from the "from math import *" line so that the
179  # python eval statement can do all the work
180  # but I could probabl;y do something cleverer by spcifying globals and locals to the
181  # eval statement (but the globals() would be needed for all the math functions)
182 
183  # first step is to substitute the genes
184  match = re.search(r'g *\‍(', insert_string)
185  if match:
186  closing_bracket = find_unmatched_close_bracket(insert_string[match.end():])
187  if closing_bracket == -1:
188  print('Error parsing genes in %s' % (insert_string))
189  sys.exit(1)
190  index = int(0.5 + parse_insert(insert_string[match.end(): match.end() + closing_bracket], genes, verbose))
191  prefix = insert_string[:match.start()]
192  if match.end() + closing_bracket + 1 < len(insert_string):
193  suffix = insert_string[match.end() + closing_bracket + 1:]
194  else:
195  suffix = ''
196  return parse_insert('%s%.18e%s' % (prefix, genes[index], suffix), genes, verbose)
197 
198  # second step is to handle functions not understood by python
199  # if(test,result_when_true,result_when_false)
200  match = re.search(r'if *\‍(', insert_string)
201  if match:
202  closing_bracket = find_unmatched_close_bracket(insert_string[match.end():])
203  if closing_bracket == -1:
204  print('Error parsing genes in %s' % (insert_string))
205  sys.exit(1)

207  arguments = split_on_unbracketed_command(argument_string)
208  if len(arguments) != 3:
209  print('if() requires 3 arguments %s' % (insert_string))

244  num_brackets = 0
245  split_string = []
246  last_string_start = 0
247  for i in range(0, len(input_string)):
248  if input_string[i] == '(':

251  num_brackets = num_brackets - 1

264 def create_smart_substitution(data):
265  smart_substitution_text_components = []
266  smart_substitution_parser_text = []
267  ptr1 = 0
268  ptr2 = data.find("[[")
269  if ptr2 == -1:
270  print("Error: could not find any [[\n")
271  sys.exit(1)
272  while True:
273  # print(ptr1, ptr2)
274  smart_substitution_text_components.append(data[ptr1: ptr2])
275  ptr2 += 2
276  ptr1 = data[ptr2:].find("]]")
277  if ptr1 == -1:
278  print("Error: could not find matching ]]\n")
279  sys.exit(1)
280  ptr1 += ptr2
281  smart_substitution_parser_text.append(data[ptr2: ptr1])
282  ptr1 += 2
283  ptr2 = data[ptr1:].find("[[")
284  if ptr2 == -1:
285  break
286  ptr2 += ptr1
287  smart_substitution_text_components.append(data[ptr1:])
288  for i in range(0, len(smart_substitution_parser_text)): # convert g[] to g()
289  smart_substitution_parser_text[i] = re.sub(r'g\[(\d+)\]', r'g(\1)', smart_substitution_parser_text[i])
290  return (smart_substitution_text_components, smart_substitution_parser_text)

292 def read_ranges(file_name):

294  with open(file_name, 'r', encoding='utf8') as f_in:
295  lines = f_in.read().splitlines()
296  if len(lines) == 0:

300  if len(column_names) == 0:
301  print('Error: could not parse column_names in "%s"' % (file_name))
302  sys.exit(1)
303  for column_name in column_names:
304  ranges[column_name] = []
305 
306  for i in range(1, len(lines)):
307  tokens = lines[i].split()
308  if len(tokens) == 0: # skip blank lines
309  continue
310  if len(tokens) != len(column_names):
311  print('Error: could not parse data in "%s"' % (file_name))
312  sys.exit(1)
313  for j in range(0, len(tokens)):
314  ranges[column_names[j]].append(float(tokens[j]))
