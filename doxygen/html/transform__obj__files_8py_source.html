246 def GetRotationTo(v0, v1):
247  # Based on Stan Melax's article in Game Programming Gems
248  #vector a = crossproduct(v1, v2)
249  #q.xyz = a
250  #q.w = sqrt((v1.Length ^ 2) * (v2.Length ^ 2)) + dotproduct(v1, v2)
251 
252  v0 = Normalise3x1(v0)
253  v1 = Normalise3x1(v1)
254 
255  d = DotProduct3x1(v0, v1)
256  # If dot == 1, vectors are the same
257  if (d >= 1.0):
258  return [1, 0, 0, 0] # identity quaternion
259 
260  if (d < (1e-7 - 1.0)):
261  # Generate an axis
262  axis = CrossProduct3x1([1, 0, 0], v0)
263  if (Magnitude3x1(axis) < 1e-6 and Magnitude3x1(axis) > -1e-6): # pick another if colinear
264  axis = CrossProduct3x1([0, 1, 0], v0)
265  axis= Normalise3x1(axis)
266  q = QuaternionFromAxisAngle(axis, math.pi)
267 
268  else:
269 
270  s = math.sqrt( (1+d)*2 )
271  invs = 1 / s
272 
273  c = CrossProduct3x1(v0, v1)
274 
275  q = [0, 0, 0, 0]
276  q[1] = c[0] * invs
277  q[2] = c[1] * invs
278  q[3] = c[2] * invs
279  q[0] = s * 0.5
280  q = Normalise4x1(q)

284 def QuaternionVectorRotate(q, v):
285 
286  QwQx = q[0] * q[1]
287  QwQy = q[0] * q[2]
288  QwQz = q[0] * q[3]
289  QxQy = q[1] * q[2]
290  QxQz = q[1] * q[3]
291  QyQz = q[2] * q[3]
292 
293  rx = 2* (v[1] * (-QwQz + QxQy) + v[2] *( QwQy + QxQz))
294  ry = 2* (v[0] * ( QwQz + QxQy) + v[2] *(-QwQx + QyQz))
295  rz = 2* (v[0] * (-QwQy + QxQz) + v[1] *( QwQx + QyQz))
296 
297  QwQw = q[0] * q[0]
298  QxQx = q[1] * q[1]
299  QyQy = q[2] * q[2]
300  QzQz = q[3] * q[3]
301 
302  rx += v[0] * (QwQw + QxQx - QyQy - QzQz)
303  ry += v[1] * (QwQw - QxQx + QyQy - QzQz)
304  rz += v[2] * (QwQw - QxQx - QyQy + QzQz)
305 
306  return [rx, ry, rz]

326 def QuaternionFromMatrix(R):
327  # quaternion is [w, x, y, z]
328  # matrix is:
329  # R.e11 R.e12 R.e13
330  # R.e21 R.e22 R.e23
331  # R.e31 R.e32 R.e33
332  q = [0,0,0,0]
333  tr = R.e11 + R.e22 + R.e33
334  if (tr >= 0):
335  s = math.sqrt (tr + 1)
336  q[0] = 0.5 * s
337  s = 0.5 * (1.0/s)
338  q[1] = (R.e32 - R.e23) * s
339  q[2] = (R.e13 - R.e31) * s
340  q[3] = (R.e21 - R.e12) * s
341  return q
342  else:
343  # find the largest diagonal element and jump to the appropriate case
344  if (R.e22 > R.e11):
345  if (R.e33 > R.e22):
346  s = math.sqrt((R.e33 - (R.e11 + R.e22)) + 1)
347  q[3] = 0.5 * s
348  s = 0.5 * (1.0/s)
349  q[1] = (R.e31 + R.e13) * s
350  q[2] = (R.e23 + R.e32) * s
351  q[0] = (R.e21 - R.e12) * s
352  return q
353  s = math.sqrt((R.e22 - (R.e33 + R.e11)) + 1)
354  q[2] = 0.5 * s
355  s = 0.5 * (1.0/s)
356  q[3] = (R.e23 + R.e32) * s
357  q[1] = (R.e12 + R.e21) * s
358  q[0] = (R.e13 - R.e31) * s
359  return q
360  if (R.e33 > R.e11):
361  s = math.sqrt((R.e33 - (R.e11 + R.e22)) + 1)
362  q[3] = 0.5 * s
363  s = 0.5 * (1.0/s)
364  q[1] = (R.e31 + R.e13) * s
365  q[2] = (R.e23 + R.e32) * s
366  q[0] = (R.e21 - R.e12) * s
367  return q
368  s = math.sqrt((R.e11 - (R.e22 + R.e33)) + 1)
369  q[1] = 0.5 * s
370  s = 0.5 * (1.0/s)
371  q[2] = (R.e12 + R.e21) * s
372  q[3] = (R.e31 + R.e13) * s
373  q[0] = (R.e32 - R.e23) * s
374  return q
